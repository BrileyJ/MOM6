38 !> Apply shortwave heating below the boundary layer (when running with the bulk mixed layer inhereted
39 !! from GOLD) or throughout the water column.
40 !!
41 !! In addition, it causes all of the remaining SW radiation to be absorbed, provided that the total
42 !! water column thickness is greater than H_limit_fluxes.
43 !! For thinner water columns, the heating is scaled down proportionately, the assumption being that the
44 !! remaining heating (which is left in Pen_SW) should go into an (absent for now) ocean bottom sediment layer.

46  adjustAbsorptionProfile, absorbAllSW, T, Pen_SW_bnd, &

52  real, dimension(:,:,:), intent(in) :: opacity_band !< Opacity in each band of penetrating
53  !! shortwave radiation [H-1 ~> m-1 or m2 kg-1].
54  !! The indicies are band, i, k.
55  integer, intent(in) :: nsw !< Number of bands of penetrating
56  !! shortwave radiation.
57  integer, intent(in) :: j !< j-index to work on.
58  real, intent(in) :: dt !< Time step [s].
59  real, intent(in) :: h_limit_fluxes !< If the total ocean depth is
60  !! less than this, they are scaled away
61  !! to avoid numerical instabilities
62  !! [H ~> m or kg m-2]. This would
63  !! not be necessary if a finite heat
64  !! capacity mud-layer were added.
65  logical, intent(in) :: adjustabsorptionprofile !< If true, apply
66  !! heating above the layers in which it
67  !! should have occurred to get the
68  !! correct mean depth (and potential
69  !! energy change) of the shortwave that
70  !! should be absorbed by each layer.
71  logical, intent(in) :: absorballsw !< If true, apply heating above the
72  !! layers in which it should have occurred
73  !! to get the correct mean depth (and
74  !! potential energy change) of the
75  !! shortwave that should be absorbed by
76  !! each layer.
77  real, dimension(SZI_(G),SZK_(G)), intent(inout) :: t !< Layer potential/conservative
78  !! temperatures [degC]
79  real, dimension(:,:), intent(inout) :: pen_sw_bnd !< Penetrating shortwave heating in
80  !! each band that hits the bottom and will
81  !! will be redistributed through the water
82  !! column [degC H ~> degC m or degC kg m-2],
83  !! size nsw x SZI_(G).
84  real, dimension(SZI_(G),SZK_(G)), optional, intent(in) :: eps !< Small thickness that must remain in
85  !! each layer, and which will not be
86  !! subject to heating [H ~> m or kg m-2]
87  integer, dimension(SZI_(G),SZK_(G)), optional, intent(in) :: ksort !< Density-sorted k-indicies.
88  real, dimension(SZI_(G)), optional, intent(in) :: htot !< Total mixed layer thickness [H ~> m or kg m-2].
89  real, dimension(SZI_(G)), optional, intent(inout) :: ttot !< Depth integrated mixed layer
90  !! temperature [degC H ~> degC m or degC kg m-2]
91  real, dimension(SZI_(G),SZK_(G)), optional, intent(in) :: dsv_dt !< The partial derivative of specific
92  !! volume with temperature [m3 kg-1 degC-1].
93  real, dimension(SZI_(G),SZK_(G)), optional, intent(inout) :: tke !< The TKE sink from mixing the heating
94  !! throughout a layer [J m-2].
95  ! Local variables
96  real, dimension(SZI_(G),SZK_(G)) :: &
97  t_chg_above ! A temperature change that will be applied to all the thick
98  ! layers above a given layer [degC]. This is only nonzero if
99  ! adjustAbsorptionProfile is true, in which case the net
100  ! change in the temperature of a layer is the sum of the
101  ! direct heating of that layer plus T_chg_above from all of
102  ! the layers below, plus any contribution from absorbing
103  ! radiation that hits the bottom.
104  real, dimension(SZI_(G)) :: &
105  h_heat, & ! The thickness of the water column that will be heated by
106  ! any remaining shortwave radiation [H ~> m or kg m-2].
107  t_chg, & ! The temperature change of thick layers due to the remaining
108  ! shortwave radiation and contributions from T_chg_above [degC].
109  pen_sw_rem ! The sum across all wavelength bands of the penetrating shortwave
110  ! heating that hits the bottom and will be redistributed through
111  ! the water column [degC H ~> degC m or degC kg m-2]
112  real :: sw_trans ! fraction of shortwave radiation that is not
113  ! absorbed in a layer [nondim]
114  real :: unabsorbed ! fraction of the shortwave radiation that
115  ! is not absorbed because the layers are too thin
116  real :: ih_limit ! inverse of the total depth at which the
117  ! surface fluxes start to be limited [H-1 ~> m-1 or m2 kg-1]
118  real :: h_min_heat ! minimum thickness layer that should get heated [H ~> m or kg m-2]
119  real :: opt_depth ! optical depth of a layer [nondim]
120  real :: exp_od ! exp(-opt_depth) [nondim]
121  real :: heat_bnd ! heating due to absorption in the current
122  ! layer by the current band, including any piece that
123  ! is moved upward [degC H ~> degC m or degC kg m-2]
124  real :: swa ! fraction of the absorbed shortwave that is
125  ! moved to layers above with adjustAbsorptionProfile [nondim]
126  real :: coswa_frac ! The fraction of SWa that is actually moved upward.
127  real :: min_sw_heating ! A minimum remaining shortwave heating rate that will be simply
128  ! absorbed in the next layer for computational efficiency, instead of
129  ! continuing to penetrate [degC H s-1 ~> degC m s-1 or degC kg m-2 s-1].
130  ! The default, 2.5e-11, is about 0.08 degC m / century.
131  real :: epsilon ! A small thickness that must remain in each
132  ! layer, and which will not be subject to heating [H ~> m or kg m-2]
133  real :: i_g_earth
134  real :: g_hconv2
135  logical :: sw_remains ! If true, some column has shortwave radiation that
136  ! was not entirely absorbed.
137  logical :: tke_calc ! If true, calculate the implications to the
138  ! TKE budget of the shortwave heating.
139  real :: c1_6, c1_60
140  integer :: is, ie, nz, i, k, ks, n
141  sw_remains = .false.
142 
143  min_sw_heating = 2.5e-11
144 
145  h_min_heat = 2.0*gv%Angstrom_H + gv%H_subroundoff
146  is = g%isc ; ie = g%iec ; nz = g%ke
147  c1_6 = 1.0 / 6.0 ; c1_60 = 1.0 / 60.0
148 
149  tke_calc = (present(tke) .and. present(dsv_dt))
150  g_hconv2 = gv%H_to_Pa * gv%H_to_kg_m2
151 
152  h_heat(:) = 0.0
153  if (present(htot)) then ; do i=is,ie ; h_heat(i) = htot(i) ; enddo ; endif
154 
155  ! Apply penetrating SW radiation to remaining parts of layers.
156  ! Excessively thin layers are not heated to avoid runaway temps.
157  do ks=1,nz ; do i=is,ie
158  k = ks
159  if (present(ksort)) then
160  if (ksort(i,ks) <= 0) cycle
161  k = ksort(i,ks)
162  endif
163  epsilon = 0.0 ; if (present(eps)) epsilon = eps(i,k)
164 
165  t_chg_above(i,k) = 0.0
166 
167  if (h(i,k) > 1.5*epsilon) then
168  do n=1,nsw ; if (pen_sw_bnd(n,i) > 0.0) then
169  ! SW_trans is the SW that is transmitted THROUGH the layer
170  opt_depth = h(i,k) * opacity_band(n,i,k)
171  exp_od = exp(-opt_depth)
172  sw_trans = exp_od
173 
174  ! Heating at a rate of less than 10-4 W m-2 = 10-3 K m / Century,
175  ! and of the layer in question less than 1 K / Century, can be
176  ! absorbed without further penetration.
177  ! ###Make these numbers into parameters!
178  if (nsw*pen_sw_bnd(n,i)*sw_trans < &
179  dt*min_sw_heating*min(gv%m_to_H,1e3*h(i,k)) ) sw_trans = 0.0
180 
181  heat_bnd = pen_sw_bnd(n,i) * (1.0 - sw_trans)
182  if (adjustabsorptionprofile .and. (h_heat(i) > 0.0)) then
183  ! In this case, a fraction of the heating is applied to the
184  ! overlying water so that the mean pressure at which the shortwave
185  ! heating occurs is exactly what it would have been with a careful
186  ! pressure-weighted averaging of the exponential heating profile,
187  ! hence there should be no TKE budget requirements due to this
188  ! layer. Very clever, but this is also limited so that the
189  ! water above is not heated at a faster rate than the layer
190  ! actually being heated, i.e., SWA <= h_heat / (h_heat + h(i,k))
191  ! and takes the energetics of the rest of the heating into account.
192  ! (-RWH, ~7 years later.)
193  if (opt_depth > 1e-5) then
194  swa = ((opt_depth + (opt_depth + 2.0)*exp_od) - 2.0) / &
195  ((opt_depth + opacity_band(n,i,k) * h_heat(i)) * &
196  (1.0 - exp_od))
197  else
198  ! Use Taylor series expansion of the expression above for a
199  ! more accurate form with very small layer optical depths.
200  swa = h(i,k) * (opt_depth * (1.0 - opt_depth)) / &
201  ((h_heat(i) + h(i,k)) * (6.0 - 3.0*opt_depth))
202  endif
203  coswa_frac = 0.0
204  if (swa*(h_heat(i) + h(i,k)) > h_heat(i)) then
205  coswa_frac = (swa*(h_heat(i) + h(i,k)) - h_heat(i) ) / &
206  (swa*(h_heat(i) + h(i,k)))
207  swa = h_heat(i) / (h_heat(i) + h(i,k))
208  endif
209 
210  t_chg_above(i,k) = t_chg_above(i,k) + (swa * heat_bnd) / h_heat(i)
211  t(i,k) = t(i,k) + ((1.0 - swa) * heat_bnd) / h(i,k)
212  else
213  coswa_frac = 1.0
214  t(i,k) = t(i,k) + pen_sw_bnd(n,i) * (1.0 - sw_trans) / h(i,k)
215  endif
216 
217  if (tke_calc) then
218  if (opt_depth > 1e-2) then
219  tke(i,k) = tke(i,k) - coswa_frac*heat_bnd*dsv_dt(i,k)* &
220  (0.5*h(i,k)*g_hconv2) * &
221  (opt_depth*(1.0+exp_od) - 2.0*(1.0-exp_od)) / (opt_depth*(1.0-exp_od))
222  else
223  ! Use Taylor series-derived approximation to the above expression
224  ! that is well behaved and more accurate when opt_depth is small.
225  tke(i,k) = tke(i,k) - coswa_frac*heat_bnd*dsv_dt(i,k)* &
226  (0.5*h(i,k)*g_hconv2) * &
227  (c1_6*opt_depth * (1.0 - c1_60*opt_depth**2))
228  endif
229  endif
230 
231  pen_sw_bnd(n,i) = pen_sw_bnd(n,i) * sw_trans
232  endif ; enddo
233  endif
234 
235  ! Add to the accumulated thickness above that could be heated.
236  ! Only layers greater than h_min_heat thick should get heated.
237  if (h(i,k) >= 2.0*h_min_heat) then
238  h_heat(i) = h_heat(i) + h(i,k)
239  elseif (h(i,k) > h_min_heat) then
240  h_heat(i) = h_heat(i) + (2.0*h(i,k) - 2.0*h_min_heat)
241  endif
242  enddo ; enddo ! i & k loops
243 
244 
245 ! if (.not.absorbAllSW .and. .not.adjustAbsorptionProfile) return
246 
247  ! Unless modified, there is no temperature change due to fluxes from the bottom.
248  do i=is,ie ; t_chg(i) = 0.0 ; enddo
249 
250  if (absorballsw) then
251  ! If there is still shortwave radiation at this point, it could go into
252  ! the bottom (with a bottom mud model), or it could be redistributed back
253  ! through the water column.
254  do i=is,ie
255  pen_sw_rem(i) = pen_sw_bnd(1,i)
256  do n=2,nsw ; pen_sw_rem(i) = pen_sw_rem(i) + pen_sw_bnd(n,i) ; enddo
257  enddo
258  do i=is,ie ; if (pen_sw_rem(i) > 0.0) sw_remains = .true. ; enddo
259 
260  ih_limit = 1.0 / h_limit_fluxes
261  do i=is,ie ; if ((pen_sw_rem(i) > 0.0) .and. (h_heat(i) > 0.0)) then
262  if (h_heat(i)*ih_limit >= 1.0) then
263  t_chg(i) = pen_sw_rem(i) / h_heat(i) ; unabsorbed = 0.0
264  else
265  t_chg(i) = pen_sw_rem(i) * ih_limit
266  unabsorbed = 1.0 - h_heat(i)*ih_limit
267  endif
268  do n=1,nsw ; pen_sw_bnd(n,i) = unabsorbed * pen_sw_bnd(n,i) ; enddo
269  endif ; enddo
270  endif ! absorbAllSW
271 
272  if (absorballsw .or. adjustabsorptionprofile) then
273  do ks=nz,1,-1 ; do i=is,ie
274  k = ks
275  if (present(ksort)) then
276  if (ksort(i,ks) <= 0) cycle
277  k = ksort(i,ks)
278  endif
279 
280  if (t_chg(i) > 0.0) then
281  ! Only layers greater than h_min_heat thick should get heated.
282  if (h(i,k) >= 2.0*h_min_heat) then ; t(i,k) = t(i,k) + t_chg(i)
283  elseif (h(i,k) > h_min_heat) then
284  t(i,k) = t(i,k) + t_chg(i) * (2.0 - 2.0*h_min_heat/h(i,k))
285  endif
286  endif
287  ! Increase the heating for layers above.
288  t_chg(i) = t_chg(i) + t_chg_above(i,k)
289  enddo ; enddo
290  if (present(htot) .and. present(ttot)) then
291  do i=is,ie ; ttot(i) = ttot(i) + t_chg(i) * htot(i) ; enddo
292  endif
293  endif ! absorbAllSW .or. adjustAbsorptionProfile
294 
295 end subroutine absorbremainingsw

363  netpen(i,k+1) = 0.

388  elseif (h(i,k) > h_min_heat) then
389  h_heat(i) = h_heat(i) + (2.0*h(i,k) - 2.0*h_min_heat)

391  enddo ! i loop
392  enddo ! k loop

397  ! the bottom (with a bottom mud model), or it could be redistributed back
398  ! through the water column.
399  do i=is,ie
400  pen_sw_rem(i) = pen_sw_bnd(1,i)
401  do n=2,nsw ; pen_sw_rem(i) = pen_sw_rem(i) + pen_sw_bnd(n,i) ; enddo
402  enddo
403  do i=is,ie ; if (pen_sw_rem(i) > 0.0) sw_remains = .true. ; enddo
404 
405  ih_limit = 1.0 / h_limit_fluxes
406  do i=is,ie ; if ((pen_sw_rem(i) > 0.0) .and. (h_heat(i) > 0.0)) then
407  if (h_heat(i)*ih_limit < 1.0) then
408  unabsorbed = 1.0 - h_heat(i)*ih_limit
409  else
410  unabsorbed = 0.0
411  endif
412  do n=1,nsw ; pen_sw_bnd(n,i) = unabsorbed * pen_sw_bnd(n,i) ; enddo
413  endif ; enddo
