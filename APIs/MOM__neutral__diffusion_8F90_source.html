232 !> Calculate remapping factors for u/v columns used to map adjoining columns to

234 subroutine neutral_diffusion_calc_coeffs(G, GV, h, T, S, CS)

236  type(verticalgrid_type), intent(in) :: gv !< ocean vertical grid structure
237  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h !< Layer thickness [H ~> m or kg m-2]
238  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: t !< Potential temperature [degC]
239  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: s !< Salinity [ppt]

241 
242  ! Local variables

247  integer :: imethod
248  real, dimension(SZI_(G)) :: ref_pres ! Reference pressure used to calculate alpha/beta
249  real :: h_neglect, h_neglect_edge
250  real :: pa_to_h
251 
252  pa_to_h = 1. / gv%H_to_pa
253 
254  !### Try replacing both of these with GV%H_subroundoff
255  if (gv%Boussinesq) then
256  h_neglect = gv%m_to_H*1.0e-30 ; h_neglect_edge = gv%m_to_H*1.0e-10
257  else
258  h_neglect = gv%kg_m2_to_H*1.0e-30 ; h_neglect_edge = gv%kg_m2_to_H*1.0e-10
259  endif
260 
261  ! If doing along isopycnal diffusion (as opposed to neutral diffusion, set the reference pressure)
262  if (cs%ref_pres>=0.) then
263  ref_pres(:) = cs%ref_pres
264  endif
265 
266  if (cs%continuous_reconstruction) then
267  cs%dRdT(:,:,:) = 0.
268  cs%dRdS(:,:,:) = 0.
269  else
270  cs%T_i(:,:,:,:) = 0.
271  cs%S_i(:,:,:,:) = 0.
272  cs%dRdT_i(:,:,:,:) = 0.
273  cs%dRdS_i(:,:,:,:) = 0.
274  cs%ns(:,:) = 0.
275  cs%stable_cell(:,:,:) = .true.
276  endif
277 
278  ! Calculate pressure at interfaces and layer averaged alpha/beta
279  cs%Pint(:,:,1) = 0.
280  do k=1,g%ke ; do j=g%jsc-1, g%jec+1 ; do i=g%isc-1,g%iec+1
281  cs%Pint(i,j,k+1) = cs%Pint(i,j,k) + h(i,j,k)*gv%H_to_Pa
282  enddo ; enddo ; enddo
283 
284  do j = g%jsc-1, g%jec+1
285  ! Interpolate state to interface
286  do i = g%isc-1, g%iec+1
287  if (cs%continuous_reconstruction) then
288  call interface_scalar(g%ke, h(i,j,:), t(i,j,:), cs%Tint(i,j,:), 2, h_neglect)
289  call interface_scalar(g%ke, h(i,j,:), s(i,j,:), cs%Sint(i,j,:), 2, h_neglect)
290  else
291  call build_reconstructions_1d( cs%remap_CS, g%ke, h(i,j,:), t(i,j,:), cs%ppoly_coeffs_T(i,j,:,:), &
292  cs%T_i(i,j,:,:), ppoly_r_s, imethod, h_neglect, h_neglect_edge )
293  call build_reconstructions_1d( cs%remap_CS, g%ke, h(i,j,:), s(i,j,:), cs%ppoly_coeffs_S(i,j,:,:), &
294  cs%S_i(i,j,:,:), ppoly_r_s, imethod, h_neglect, h_neglect_edge )
295  endif
296  enddo
297 
298  ! Continuous reconstruction
299  if (cs%continuous_reconstruction) then
300  do k = 1, g%ke+1
301  if (cs%ref_pres<0) ref_pres(:) = cs%Pint(:,j,k)
302  call calculate_density_derivs(cs%Tint(:,j,k), cs%Sint(:,j,k), ref_pres, &
303  cs%dRdT(:,j,k), cs%dRdS(:,j,k), g%isc-1, g%iec-g%isc+3, cs%EOS)
304  enddo
305  else ! Discontinuous reconstruction
306  do k = 1, g%ke
307  if (cs%ref_pres<0) ref_pres(:) = cs%Pint(:,j,k)
308  call calculate_density_derivs(cs%T_i(:,j,k,1), cs%S_i(:,j,k,1), ref_pres, &
309  cs%dRdT_i(:,j,k,1), cs%dRdS_i(:,j,k,1), g%isc-1, g%iec-g%isc+3, cs%EOS)
310  if (cs%ref_pres<0) then
311  ref_pres(:) = cs%Pint(:,j,k+1)
312  endif
313  call calculate_density_derivs(cs%T_i(:,j,k,2), cs%S_i(:,j,k,2), ref_pres, &
314  cs%dRdT_i(:,j,k,2), cs%dRdS_i(:,j,k,2), g%isc-1, g%iec-g%isc+3, cs%EOS)
315  enddo
316  endif
317  enddo
318 
319  if (.not. cs%continuous_reconstruction) then
320  do j = g%jsc-1, g%jec+1 ; do i = g%isc-1, g%iec+1
321  call mark_unstable_cells( g%ke, cs%dRdT_i(i,j,:,:), cs%dRdS_i(i,j,:,:), cs%T_i(i,j,:,:), cs%S_i(i,j,:,:), &
322  cs%stable_cell(i,j,:), cs%ns(i,j) )
323  enddo ; enddo
324  endif
325 
326  cs%uhEff(:,:,:) = 0.
327  cs%vhEff(:,:,:) = 0.
328  cs%uPoL(:,:,:) = 0.
329  cs%vPoL(:,:,:) = 0.
330  cs%uPoR(:,:,:) = 0.
331  cs%vPoR(:,:,:) = 0.
332  cs%uKoL(:,:,:) = 1
333  cs%vKoL(:,:,:) = 1
334  cs%uKoR(:,:,:) = 1
335  cs%vKoR(:,:,:) = 1
336 
337  ! Neutral surface factors at U points
338  do j = g%jsc, g%jec ; do i = g%isc-1, g%iec
339  if (g%mask2dCu(i,j) > 0.) then
340  if (cs%continuous_reconstruction) then
341  call find_neutral_surface_positions_continuous(g%ke, &
342  cs%Pint(i,j,:), cs%Tint(i,j,:), cs%Sint(i,j,:), cs%dRdT(i,j,:), cs%dRdS(i,j,:), &
343  cs%Pint(i+1,j,:), cs%Tint(i+1,j,:), cs%Sint(i+1,j,:), cs%dRdT(i+1,j,:), cs%dRdS(i+1,j,:), &
344  cs%uPoL(i,j,:), cs%uPoR(i,j,:), cs%uKoL(i,j,:), cs%uKoR(i,j,:), cs%uhEff(i,j,:) )
345  else
346  call find_neutral_surface_positions_discontinuous(cs, g%ke, cs%ns(i,j)+cs%ns(i+1,j), &
347  cs%Pint(i,j,:), h(i,j,:), cs%T_i(i,j,:,:), cs%S_i(i,j,:,:), &
348  cs%dRdT_i(i,j,:,:), cs%dRdS_i(i,j,:,:), cs%stable_cell(i,j,:), &
349  cs%Pint(i+1,j,:), h(i+1,j,:), cs%T_i(i+1,j,:,:), cs%S_i(i+1,j,:,:), &
350  cs%dRdT_i(i+1,j,:,:), cs%dRdS_i(i+1,j,:,:), cs%stable_cell(i+1,j,:), &
351  cs%uPoL(i,j,:), cs%uPoR(i,j,:), cs%uKoL(i,j,:), cs%uKoR(i,j,:), cs%uhEff(i,j,:), &
352  cs%ppoly_coeffs_T(i,j,:,:), cs%ppoly_coeffs_S(i,j,:,:), &
353  cs%ppoly_coeffs_T(i+1,j,:,:), cs%ppoly_coeffs_S(i+1,j,:,:))
354  endif
355  endif
356  enddo ; enddo
357 
358  ! Neutral surface factors at V points
359  do j = g%jsc-1, g%jec ; do i = g%isc, g%iec
360  if (g%mask2dCv(i,j) > 0.) then

608 !> Returns the PPM quasi-fourth order edge value at k+1/2 following
609 !! equation 1.6 in Colella & Woodward, 1984: JCP 54, 174-201.
610 real function ppm_edge(hkm1, hk, hkp1, hkp2, Ak, Akp1, Pk, Pkp1, h_neglect)
611  real, intent(in) :: hkm1 !< Width of cell k-1
612  real, intent(in) :: hk !< Width of cell k
613  real, intent(in) :: hkp1 !< Width of cell k+1
614  real, intent(in) :: hkp2 !< Width of cell k+2
615  real, intent(in) :: ak !< Average scalar value of cell k
616  real, intent(in) :: akp1 !< Average scalar value of cell k+1
617  real, intent(in) :: pk !< PLM slope for cell k
618  real, intent(in) :: pkp1 !< PLM slope for cell k+1
619  real, intent(in) :: h_neglect !< A negligibly small thickness [H ~> m or kg m-2]
620 
621  ! Local variables
622  real :: r_hk_hkp1, r_2hk_hkp1, r_hk_2hkp1, f1, f2, f3, f4
623 
624  r_hk_hkp1 = hk + hkp1
625  if (r_hk_hkp1 <= 0.) then
626  ppm_edge = 0.5 * ( ak + akp1 )
627  return
628  endif
629  r_hk_hkp1 = 1. / r_hk_hkp1
630  if (hk<hkp1) then
631  ppm_edge = ak + ( hk * r_hk_hkp1 ) * ( akp1 - ak )
632  else
633  ppm_edge = akp1 + ( hkp1 * r_hk_hkp1 ) * ( ak - akp1 )
634  endif
635 
636  r_2hk_hkp1 = 1. / ( ( 2. * hk + hkp1 ) + h_neglect )
637  r_hk_2hkp1 = 1. / ( ( hk + 2. * hkp1 ) + h_neglect )
638  f1 = 1./ ( ( hk + hkp1) + ( hkm1 + hkp2 ) )
639  f2 = 2. * ( hkp1 * hk ) * r_hk_hkp1 * &
640  ( ( hkm1 + hk ) * r_2hk_hkp1 - ( hkp2 + hkp1 ) * r_hk_2hkp1 )
641  f3 = hk * ( hkm1 + hk ) * r_2hk_hkp1
642  f4 = hkp1 * ( hkp1 + hkp2 ) * r_hk_2hkp1
643 
644  ppm_edge = ppm_edge + f1 * ( f2 * ( akp1 - ak ) - ( f3 * pkp1 - f4 * pk ) )
645 
646 end function ppm_edge

686 !> Returns PLM slopes for a column where the slopes are the difference in value across each cell.
687 !! The limiting follows equation 1.8 in Colella & Woodward, 1984: JCP 54, 174-201.
688 subroutine plm_diff(nk, h, S, c_method, b_method, diff)
689  integer, intent(in) :: nk !< Number of levels
690  real, dimension(nk), intent(in) :: h !< Layer thickness [H ~> m or kg m-2]
691  real, dimension(nk), intent(in) :: S !< Layer salinity (conc, e.g. ppt)
692  integer, intent(in) :: c_method !< Method to use for the centered difference
693  integer, intent(in) :: b_method !< =1, use PCM in first/last cell, =2 uses linear extrapolation
694  real, dimension(nk), intent(inout) :: diff !< Scalar difference across layer (conc, e.g. ppt)
695  !! determined by the following values for c_method:
696  !! 1. Second order finite difference (not recommended)
697  !! 2. Second order finite volume (used in original PPM)
698  !! 3. Finite-volume weighted least squares linear fit
699  !! \todo The use of c_method to choose a scheme is inefficient
700  !! and should eventually be moved up the call tree.
701 
702  ! Local variables
703  integer :: k
704  real :: hkm1, hk, hkp1, Skm1, Sk, Skp1, diff_l, diff_r, diff_c
705 
706  do k = 2, nk-1
707  hkm1 = h(k-1)
708  hk = h(k)
709  hkp1 = h(k+1)
710 
711  if ( ( hkp1 + hk ) * ( hkm1 + hk ) > 0.) then
712  skm1 = s(k-1)
713  sk = s(k)
714  skp1 = s(k+1)
715  if (c_method==1) then
716  ! Simple centered diff (from White)
717  if ( hk + 0.5 * (hkm1 + hkp1) /= 0. ) then
718  diff_c = ( skp1 - skm1 ) * ( hk / ( hk + 0.5 * (hkm1 + hkp1) ) )
719  else
720  diff_c = 0.
721  endif
722  elseif (c_method==2) then
723  ! Second order accurate centered FV slope (from Colella and Woodward, JCP 1984)
724  diff_c = fv_diff(hkm1, hk, hkp1, skm1, sk, skp1)
725  elseif (c_method==3) then
726  ! Second order accurate finite-volume least squares slope
727  diff_c = hk * fvlsq_slope(hkm1, hk, hkp1, skm1, sk, skp1)
728  endif
729  ! Limit centered slope by twice the side differenced slopes
730  diff_l = 2. * ( sk - skm1 )
731  diff_r = 2. * ( skp1 - sk )
732  if ( signum(1., diff_l) * signum(1., diff_r) <= 0. ) then
733  diff(k) = 0. ! PCM for local extrema
734  else
735  diff(k) = sign( min( abs(diff_l), abs(diff_c), abs(diff_r) ), diff_c )
736  endif
737  else
738  diff(k) = 0. ! PCM next to vanished layers
739  endif
740  enddo
741  if (b_method==1) then ! PCM for top and bottom layer
742  diff(1) = 0.
743  diff(nk) = 0.
744  elseif (b_method==2) then ! Linear extrapolation for top and bottom interfaces
745  diff(1) = ( s(2) - s(1) ) * 2. * ( h(1) / ( h(1) + h(2) ) )
746  diff(nk) = s(nk) - s(nk-1) * 2. * ( h(nk) / ( h(nk-1) + h(nk) ) )
747  endif
748 
749 end subroutine plm_diff

751 !> Returns the cell-centered second-order finite volume (unlimited PLM) slope
752 !! using three consecutive cell widths and average values. Slope is returned
753 !! as a difference across the central cell (i.e. units of scalar S).
754 !! Discretization follows equation 1.7 in Colella & Woodward, 1984: JCP 54, 174-201.
755 real function fv_diff(hkm1, hk, hkp1, Skm1, Sk, Skp1)
756  real, intent(in) :: hkm1 !< Left cell width
757  real, intent(in) :: hk !< Center cell width
758  real, intent(in) :: hkp1 !< Right cell width
759  real, intent(in) :: skm1 !< Left cell average value

761  real, intent(in) :: skp1 !< Right cell average value
762 
763  ! Local variables
764  real :: h_sum, hp, hm
765 
766  h_sum = ( hkm1 + hkp1 ) + hk
767  if (h_sum /= 0.) h_sum = 1./ h_sum
768  hm = hkm1 + hk
769  if (hm /= 0.) hm = 1./ hm
770  hp = hkp1 + hk
771  if (hp /= 0.) hp = 1./ hp
772  fv_diff = ( hk * h_sum ) * &
773  ( ( 2. * hkm1 + hk ) * hp * ( skp1 - sk ) &
774  + ( 2. * hkp1 + hk ) * hm * ( sk - skm1 ) )

779 !! using three consecutive cell widths and average values. Slope is returned
780 !! as a gradient (i.e. units of scalar S over width units).
781 real function fvlsq_slope(hkm1, hk, hkp1, Skm1, Sk, Skp1)
782  real, intent(in) :: hkm1 !< Left cell width
783  real, intent(in) :: hk !< Center cell width
784  real, intent(in) :: hkp1 !< Right cell width
785  real, intent(in) :: skm1 !< Left cell average value
786  real, intent(in) :: sk !< Center cell average value
787  real, intent(in) :: skp1 !< Right cell average value
788 
789  ! Local variables
790  real :: xkm1, xkp1
791  real :: h_sum, hx_sum, hxsq_sum, hxy_sum, hy_sum, det
792 
793  xkm1 = -0.5 * ( hk + hkm1 )
794  xkp1 = 0.5 * ( hk + hkp1 )
795  h_sum = ( hkm1 + hkp1 ) + hk
796  hx_sum = hkm1*xkm1 + hkp1*xkp1
797  hxsq_sum = hkm1*(xkm1**2) + hkp1*(xkp1**2)
798  hxy_sum = hkm1*xkm1*skm1 + hkp1*xkp1*skp1
799  hy_sum = ( hkm1*skm1 + hkp1*skp1 ) + hk*sk
800  det = h_sum * hxsq_sum - hx_sum**2
801  if (det /= 0.) then
802  !a = ( hxsq_sum * hy_sum - hx_sum*hxy_sum ) / det ! a would be mean of straight line fit
803  fvlsq_slope = ( h_sum * hxy_sum - hx_sum*hy_sum ) / det ! Gradient of straight line fit
804  else
805  fvlsq_slope = 0. ! Adcroft's reciprocal rule
806  endif
807 end function fvlsq_slope

931  + ( drdsr(krm1+1) + drdsl(kl) ) * ( sr(krm1+1) - sl(kl) ) )
932 
933  ! Because we are looking right, the left surface, kl, is lighter than krm1+1 and should be denser than krm1
934  ! unless we are still at the top of the right column (kr=1)
935  if (drhotop >= 0. .or. kr+kl==2) then
936  por(k_surface) = 0. ! The left surface is lighter than anything in layer krm1
937  elseif (drhotop >= drhobot) then ! Right layer is unstratified
938  por(k_surface) = 1.
939  else
940  ! Linearly interpolate for the position between Pr(kr-1) and Pr(kr) where the density difference
941  ! between right and left is zero.
942  por(k_surface) = interpolate_for_nondim_position( drhotop, pr(krm1), drhobot, pr(krm1+1) )
943  endif
944  if (por(k_surface)>=1. .and. krm1<nk) then ! >= is really ==, when PoR==1 we point to the bottom of the cell
945  krm1 = krm1 + 1
946  por(k_surface) = por(k_surface) - 1.
947  endif
948  if (real(krm1-lastk_right)+(por(k_surface)-lastp_right)<0.) then
949  por(k_surface) = lastp_right
950  krm1 = lastk_right
951  endif
952  kor(k_surface) = krm1
953  if (kl <= nk) then
954  pol(k_surface) = 0.
955  kol(k_surface) = kl
956  else
957  pol(k_surface) = 1.
958  kol(k_surface) = nk
959  endif
960  if (kl <= nk) then
961  kl = kl + 1
962  else
963  reached_bottom = .true.
964  searching_right_column = .false.
965  searching_left_column = .true.
966  endif
967  else
968  stop 'Else what?'
969  endif
970 
971  lastk_left = kol(k_surface) ; lastp_left = pol(k_surface)
972  lastk_right = kor(k_surface) ; lastp_right = por(k_surface)
973 
974  ! Effective thickness
975  ! NOTE: This would be better expressed in terms of the layers thicknesses rather
976  ! than as differences of position - AJA
977  if (k_surface>1) then
978  hl = absolute_position(nk,ns,pl,kol,pol,k_surface) - absolute_position(nk,ns,pl,kol,pol,k_surface-1)
979  hr = absolute_position(nk,ns,pr,kor,por,k_surface) - absolute_position(nk,ns,pr,kor,por,k_surface-1)
980  if ( hl + hr > 0.) then
981  heff(k_surface-1) = 2. * hl * hr / ( hl + hr ) ! Harmonic mean of layer thicknesses
982  else
983  heff(k_surface-1) = 0.
984  endif
985  endif
986 
987  enddo neutral_surfaces
