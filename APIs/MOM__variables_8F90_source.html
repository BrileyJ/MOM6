297 !> Allocates the fields for the surface (return) properties of
298 !! the ocean model. Unused fields are unallocated.
299 subroutine allocate_surface_state(sfc_state, G, use_temperature, do_integrals, &
300  gas_fields_ocn, use_meltpot)
301  type(ocean_grid_type), intent(in) :: g !< ocean grid structure
302  type(surface), intent(inout) :: sfc_state !< ocean surface state type to be allocated.
303  logical, optional, intent(in) :: use_temperature !< If true, allocate the space for thermodynamic variables.
304  logical, optional, intent(in) :: do_integrals !< If true, allocate the space for vertically
305  !! integrated fields.
306  type(coupler_1d_bc_type), &
307  optional, intent(in) :: gas_fields_ocn !< If present, this type describes the ocean
308  !! ocean and surface-ice fields that will participate
309  !! in the calculation of additional gas or other
310  !! tracer fluxes, and can be used to spawn related
311  !! internal variables in the ice model.
312  logical, optional, intent(in) :: use_meltpot !< If true, allocate the space for melt potential
313 
314  ! local variables
315  logical :: use_temp, alloc_integ, use_melt_potential
316  integer :: is, ie, js, je, isd, ied, jsd, jed
317  integer :: isdb, iedb, jsdb, jedb
318 
319  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec
320  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
321  isdb = g%isdB ; iedb = g%iedB; jsdb = g%jsdB ; jedb = g%jedB
322 
323  use_temp = .true. ; if (present(use_temperature)) use_temp = use_temperature
324  alloc_integ = .true. ; if (present(do_integrals)) alloc_integ = do_integrals
325  use_melt_potential = .false. ; if (present(use_meltpot)) use_melt_potential = use_meltpot
326 
327  if (sfc_state%arrays_allocated) return
328 
329  if (use_temp) then
330  allocate(sfc_state%SST(isd:ied,jsd:jed)) ; sfc_state%SST(:,:) = 0.0
331  allocate(sfc_state%SSS(isd:ied,jsd:jed)) ; sfc_state%SSS(:,:) = 0.0
332  else
333  allocate(sfc_state%sfc_density(isd:ied,jsd:jed)) ; sfc_state%sfc_density(:,:) = 0.0
334  endif
335  allocate(sfc_state%sea_lev(isd:ied,jsd:jed)) ; sfc_state%sea_lev(:,:) = 0.0
336  allocate(sfc_state%Hml(isd:ied,jsd:jed)) ; sfc_state%Hml(:,:) = 0.0
337  allocate(sfc_state%u(isdb:iedb,jsd:jed)) ; sfc_state%u(:,:) = 0.0
338  allocate(sfc_state%v(isd:ied,jsdb:jedb)) ; sfc_state%v(:,:) = 0.0
339 
340  if (use_melt_potential) then
341  allocate(sfc_state%melt_potential(isd:ied,jsd:jed)) ; sfc_state%melt_potential(:,:) = 0.0
342  endif
343 
344  if (alloc_integ) then
345  ! Allocate structures for the vertically integrated ocean_mass, ocean_heat, and ocean_salt.
346  allocate(sfc_state%ocean_mass(isd:ied,jsd:jed)) ; sfc_state%ocean_mass(:,:) = 0.0
347  if (use_temp) then
348  allocate(sfc_state%ocean_heat(isd:ied,jsd:jed)) ; sfc_state%ocean_heat(:,:) = 0.0
349  allocate(sfc_state%ocean_salt(isd:ied,jsd:jed)) ; sfc_state%ocean_salt(:,:) = 0.0
350  endif
351  allocate(sfc_state%salt_deficit(isd:ied,jsd:jed)) ; sfc_state%salt_deficit(:,:) = 0.0
352  endif
353 
354  if (present(gas_fields_ocn)) &
355  call coupler_type_spawn(gas_fields_ocn, sfc_state%tr_fields, &
356  (/is,is,ie,ie/), (/js,js,je,je/), as_needed=.true.)
357 
358  sfc_state%arrays_allocated = .true.
359 
360 end subroutine allocate_surface_state

362 !> Deallocates the elements of a surface state type.
363 subroutine deallocate_surface_state(sfc_state)
364  type(surface), intent(inout) :: sfc_state !< ocean surface state type to be deallocated here.
365 
366  if (.not.sfc_state%arrays_allocated) return
367 
368  if (allocated(sfc_state%melt_potential)) deallocate(sfc_state%melt_potential)
369  if (allocated(sfc_state%SST)) deallocate(sfc_state%SST)
370  if (allocated(sfc_state%SSS)) deallocate(sfc_state%SSS)
371  if (allocated(sfc_state%sfc_density)) deallocate(sfc_state%sfc_density)
372  if (allocated(sfc_state%sea_lev)) deallocate(sfc_state%sea_lev)
373  if (allocated(sfc_state%Hml)) deallocate(sfc_state%Hml)
374  if (allocated(sfc_state%u)) deallocate(sfc_state%u)
375  if (allocated(sfc_state%v)) deallocate(sfc_state%v)
