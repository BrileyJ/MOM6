52 !> Add Kelvin wave to OBC registry.
53 function register_kelvin_obc(param_file, CS, OBC_Reg)
54  type(param_file_type), intent(in) :: param_file !< parameter file.
55  type(kelvin_obc_cs), pointer :: cs !< Kelvin wave control structure.
56  type(obc_registry_type), pointer :: obc_reg !< OBC registry.
57  logical :: register_kelvin_obc
58  character(len=40) :: mdl = "register_Kelvin_OBC" !< This subroutine's name.
59  character(len=32) :: casename = "Kelvin wave" !< This case's name.
60  character(len=200) :: config
61 
62  if (associated(cs)) then
63  call mom_error(warning, "register_Kelvin_OBC called with an "// &
64  "associated control structure.")
65  return
66  endif
67  allocate(cs)
68 
69  call log_version(param_file, mdl, version, "")
70  call get_param(param_file, mdl, "KELVIN_WAVE_MODE", cs%mode, &
71  "Vertical Kelvin wave mode imposed at upstream open boundary.", &
72  default=0)
73  call get_param(param_file, mdl, "F_0", cs%F_0, &
74  default=0.0, do_not_log=.true.)
75  call get_param(param_file, mdl, "TOPO_CONFIG", config, do_not_log=.true.)
76  if (trim(config) == "Kelvin") then
77  call get_param(param_file, mdl, "ROTATED_COAST_OFFSET_1", cs%coast_offset1, &
78  "The distance along the southern and northern boundaries "//&
79  "at which the coasts angle in.", &
80  units="km", default=100.0)
81  call get_param(param_file, mdl, "ROTATED_COAST_OFFSET_2", cs%coast_offset2, &
82  "The distance from the southern and northern boundaries "//&
83  "at which the coasts angle in.", &
84  units="km", default=10.0)
85  call get_param(param_file, mdl, "ROTATED_COAST_ANGLE", cs%coast_angle, &
86  "The angle of the southern bondary beyond X=ROTATED_COAST_OFFSET.", &
87  units="degrees", default=11.3)
88  cs%coast_angle = cs%coast_angle * (atan(1.0)/45.) ! Convert to radians
89  cs%coast_offset1 = cs%coast_offset1 * 1.e3 ! Convert to m
90  cs%coast_offset2 = cs%coast_offset2 * 1.e3 ! Convert to m
91  endif
92  if (cs%mode /= 0) then
93  call get_param(param_file, mdl, "DENSITY_RANGE", cs%rho_range, &
94  default=2.0, do_not_log=.true.)
95  call get_param(param_file, mdl, "RHO_0", cs%rho_0, &
96  default=1035.0, do_not_log=.true.)
97  call get_param(param_file, mdl, "MAXIMUM_DEPTH", cs%H0, &
98  default=1000.0, do_not_log=.true.)
99  endif
100 
101  ! Register the Kelvin open boundary.
102  call register_obc(casename, param_file, obc_reg)
103  register_kelvin_obc = .true.
104 
105 end function register_kelvin_obc

117 !> This subroutine sets up the Kelvin topography and land mask
118 subroutine kelvin_initialize_topography(D, G, param_file, max_depth, US)
119  type(dyn_horgrid_type), intent(in) :: g !< The dynamic horizontal grid type
120  real, dimension(G%isd:G%ied,G%jsd:G%jed), &
121  intent(out) :: d !< Ocean bottom depth in m or Z if US is present
122  type(param_file_type), intent(in) :: param_file !< Parameter file structure
123  real, intent(in) :: max_depth !< Maximum model depth in the units of D
124  type(unit_scale_type), optional, intent(in) :: us !< A dimensional unit scaling type
125 
126  ! Local variables
127  character(len=40) :: mdl = "Kelvin_initialize_topography" ! This subroutine's name.
128  real :: m_to_z ! A dimensional rescaling factor.
129  real :: min_depth ! The minimum and maximum depths [Z ~> m].
130  real :: pi ! 3.1415...
131  real :: coast_offset1, coast_offset2, coast_angle, right_angle
132  integer :: i, j
133 
134  call mom_mesg(" Kelvin_initialization.F90, Kelvin_initialize_topography: setting topography", 5)
135 
136  m_to_z = 1.0 ; if (present(us)) m_to_z = us%m_to_Z
137 
138  call get_param(param_file, mdl, "MINIMUM_DEPTH", min_depth, &
139  "The minimum depth of the ocean.", units="m", default=0.0, scale=m_to_z)
140  call get_param(param_file, mdl, "ROTATED_COAST_OFFSET_1", coast_offset1, &
141  default=100.0, do_not_log=.true.)
142  call get_param(param_file, mdl, "ROTATED_COAST_OFFSET_2", coast_offset2, &
143  default=10.0, do_not_log=.true.)
144  call get_param(param_file, mdl, "ROTATED_COAST_ANGLE", coast_angle, &
145  default=11.3, do_not_log=.true.)
146 
147  coast_angle = coast_angle * (atan(1.0)/45.) ! Convert to radians
148  right_angle = 2 * atan(1.0)
149 
150  do j=g%jsc,g%jec ; do i=g%isc,g%iec
151  d(i,j) = max_depth
152  ! Southern side
153  if ((g%geoLonT(i,j) - g%west_lon > coast_offset1) .AND. &
154  (atan2(g%geoLatT(i,j) - g%south_lat + coast_offset2, &
155  g%geoLonT(i,j) - g%west_lon - coast_offset1) < coast_angle)) &
156  d(i,j) = 0.5*min_depth
157  ! Northern side
158  if ((g%geoLonT(i,j) - g%west_lon < g%len_lon - coast_offset1) .AND. &
159  (atan2(g%len_lat + g%south_lat + coast_offset2 - g%geoLatT(i,j), &
160  g%len_lon + g%west_lon - coast_offset1 - g%geoLonT(i,j)) < coast_angle)) &
161  d(i,j) = 0.5*min_depth
162 
163  if (d(i,j) > max_depth) d(i,j) = max_depth
164  if (d(i,j) < min_depth) d(i,j) = 0.5*min_depth
165  enddo ; enddo
166 
167 end subroutine kelvin_initialize_topography

169 !> This subroutine sets the properties of flow at open boundary conditions.
170 subroutine kelvin_set_obc_data(OBC, CS, G, GV, US, h, Time)
171  type(ocean_obc_type), pointer :: obc !< This open boundary condition type specifies
172  !! whether, where, and what open boundary
173  !! conditions are used.
174  type(kelvin_obc_cs), pointer :: cs !< Kelvin wave control structure.
175  type(ocean_grid_type), intent(in) :: g !< The ocean's grid structure.
176  type(verticalgrid_type), intent(in) :: gv !< The ocean's vertical grid structure.
177  type(unit_scale_type), intent(in) :: us !< A dimensional unit scaling type
178  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h !< layer thickness [H ~> m or kg m-2].
179  type(time_type), intent(in) :: time !< model time.
180 
181  ! The following variables are used to set up the transport in the Kelvin example.
182  real :: time_sec, cff
183  real :: n0 ! Brunt-Vaisala frequency [s-1]
184  real :: plx !< Longshore wave parameter
185  real :: pmz !< Vertical wave parameter
186  real :: lambda !< Offshore decay scale
187  real :: omega !< Wave frequency [s-1]
188  real :: pi
189  integer :: i, j, k, n, is, ie, js, je, isd, ied, jsd, jed, nz
190  integer :: isdb, iedb, jsdb, jedb
191  real :: fac, x, y, x1, y1
192  real :: val1, val2, sina, cosa
193  type(obc_segment_type), pointer :: segment => null()
194 
195  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
196  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
197  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
198 
199  if (.not.associated(obc)) call mom_error(fatal, 'Kelvin_initialization.F90: '// &
200  'Kelvin_set_OBC_data() was called but OBC type was not initialized!')
201 
202  time_sec = time_type_to_real(time)
203  pi = 4.0*atan(1.0)
204  fac = 1.0
205 
206  if (cs%mode == 0) then
207  omega = 2.0 * pi / (12.42 * 3600.0) ! M2 Tide period
208  val1 = us%m_to_Z * sin(omega * time_sec)
209  else
210  n0 = sqrt((cs%rho_range / cs%rho_0) * gv%g_Earth * (us%m_to_Z * cs%H0))
211  ! Two wavelengths in domain
212  plx = 4.0 * pi / g%len_lon
213  pmz = pi * cs%mode / cs%H0
214  lambda = pmz * cs%F_0 / n0
215  omega = cs%F_0 * plx / lambda
216 
217  ! lambda = PI * CS%mode * CS%F_0 / (CS%H0 * N0)
218  ! omega = (4.0 * CS%H0 * N0) / (CS%mode * G%len_lon)
219  endif
220 
221  sina = sin(cs%coast_angle)
222  cosa = cos(cs%coast_angle)
223  do n=1,obc%number_of_segments
224  segment => obc%segment(n)
225  if (.not. segment%on_pe) cycle
226  ! Apply values to the inflow end only.
227  if (segment%direction == obc_direction_e) cycle
228  if (segment%direction == obc_direction_n) cycle
229 
230  ! This should be somewhere else...
231  segment%Velocity_nudging_timescale_in = 1.0/(0.3*86400)
232 
233  if (segment%direction == obc_direction_w) then
234  isdb = segment%HI%IsdB ; iedb = segment%HI%IedB
235  jsd = segment%HI%jsd ; jed = segment%HI%jed
236  jsdb = segment%HI%JsdB ; jedb = segment%HI%JedB
237  do j=jsd,jed ; do i=isdb,iedb
238  x1 = 1000. * g%geoLonCu(i,j)
239  y1 = 1000. * g%geoLatCu(i,j)
240  x = (x1 - cs%coast_offset1) * cosa + y1 * sina
241  y = - (x1 - cs%coast_offset1) * sina + y1 * cosa
242  if (cs%mode == 0) then
243  cff = sqrt(gv%g_Earth * 0.5 * (g%bathyT(i+1,j) + g%bathyT(i,j)))
244  val2 = fac * exp(- cs%F_0 * y / cff)
245  segment%eta(i,j) = val2 * cos(omega * time_sec)
246  segment%normal_vel_bt(i,j) = val1 * cff * cosa / &
247  (0.5 * (g%bathyT(i+1,j) + g%bathyT(i,j))) * val2
248  else
249  ! Not rotated yet
250  segment%eta(i,j) = 0.0
251  segment%normal_vel_bt(i,j) = 0.0
252  if (segment%nudged) then
253  do k=1,nz
254  segment%nudged_normal_vel(i,j,k) = fac * lambda / cs%F_0 * &
255  exp(- lambda * y) * cos(pi * cs%mode * (k - 0.5) / nz) * &
256  cos(omega * time_sec)
257  enddo
258  elseif (segment%specified) then
259  do k=1,nz
260  segment%normal_vel(i,j,k) = fac * lambda / cs%F_0 * &
261  exp(- lambda * y) * cos(pi * cs%mode * (k - 0.5) / nz) * &
262  cos(omega * time_sec)
263  segment%normal_trans(i,j,k) = segment%normal_vel(i,j,k) * &
264  h(i+1,j,k) * g%dyCu(i,j)
265  enddo
266  endif
267  endif
268  enddo ; enddo
269  if (associated(segment%tangential_vel)) then
270  do j=jsdb+1,jedb-1 ; do i=isdb,iedb
271  x1 = 1000. * g%geoLonBu(i,j)
272  y1 = 1000. * g%geoLatBu(i,j)
273  x = (x1 - cs%coast_offset1) * cosa + y1 * sina
274  y = - (x1 - cs%coast_offset1) * sina + y1 * cosa
275  !### Problem: val2 & cff could be a functions of space, but are not set in this loop.
276  !### Problem: Is val2 in the numerator or denominator below?
277  if (cs%mode == 0) then
278  do k=1,nz
279  segment%tangential_vel(i,j,k) = val1 * cff * sina / &
280  (0.25 * (g%bathyT(i+1,j) + g%bathyT(i,j) + &
281  g%bathyT(i+1,j+1) + g%bathyT(i,j+1))) * val2
282 !### For rotational symmetry, this should be:
283 ! segment%tangential_vel(I,J,k) = val1 * cff * sina / &
284 ! ( 0.25*((G%bathyT(i,j) + G%bathyT(i+1,j+1)) +&
285 ! (G%bathyT(i+1,j) + G%bathyT(i,j+1))) ) * val2
286  enddo
287  endif
288  enddo ; enddo
289  endif
290  else
291  isd = segment%HI%isd ; ied = segment%HI%ied
292  jsdb = segment%HI%JsdB ; jedb = segment%HI%JedB
293  do j=jsdb,jedb ; do i=isd,ied
294  x1 = 1000. * g%geoLonCv(i,j)
295  y1 = 1000. * g%geoLatCv(i,j)
296  x = (x1 - cs%coast_offset1) * cosa + y1 * sina
297  y = - (x1 - cs%coast_offset1) * sina + y1 * cosa
298  if (cs%mode == 0) then
299  cff = sqrt(gv%g_Earth * 0.5 * (g%bathyT(i,j+1) + g%bathyT(i,j)))
300  val2 = fac * exp(- 0.5 * us%s_to_T * (g%CoriolisBu(i,j) + g%CoriolisBu(i-1,j)) * y / cff)
301  segment%eta(i,j) = val2 * cos(omega * time_sec)
302  segment%normal_vel_bt(i,j) = val1 * cff * sina / &
303  (0.5*(g%bathyT(i+1,j) + g%bathyT(i,j))) * val2
304  else
305  ! Not rotated yet
306  segment%eta(i,j) = 0.0
307  segment%normal_vel_bt(i,j) = 0.0
308  if (segment%nudged) then
309  do k=1,nz
310  segment%nudged_normal_vel(i,j,k) = fac * lambda / cs%F_0 * &
311  exp(- lambda * y) * cos(pi * cs%mode * (k - 0.5) / nz) * cosa
312  enddo
313  elseif (segment%specified) then
314  do k=1,nz
315  segment%normal_vel(i,j,k) = fac * lambda / cs%F_0 * &
316  exp(- lambda * y) * cos(pi * cs%mode * (k - 0.5) / nz) * cosa
317  segment%normal_trans(i,j,k) = segment%normal_vel(i,j,k) * &
318  h(i,j+1,k) * g%dxCv(i,j)
319  enddo
320  endif
