198 !> 3D advection is done by doing flux-limited nonlinear horizontal advection interspersed with an ALE

204  real, intent(in) :: time_interval !< time interval

220  ! about the vertical grid

235  real :: inum_iter

237  integer :: i, j, k, m, is, ie, js, je, isd, ied, jsd, jed, nz
238  integer :: isv, iev, jsv, jev ! The valid range of the indices.

250  gv => cs%GV

253

255  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = gv%ke
256  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
257  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
258

261  minimum_forcing_depth = cs%minimum_forcing_depth
262

264  inum_iter = 1./real(niter)
265  dt_iter = dt_offline*inum_iter
266 
267  ! Initialize working arrays
268  h_new(:,:,:) = 0.0
269  h_vol(:,:,:) = 0.0
270  uhtr_sub(:,:,:) = 0.0
271  vhtr_sub(:,:,:) = 0.0
272 
273  ! converged should only be true if there are no remaining mass fluxes
274  converged = .false.
275 
276  ! Tracers are transported using the stored mass fluxes. Where possible, operators are Strang-split around
277  ! the call to
278  ! 1) Using the layer thicknesses and tracer concentrations from the previous timestep,
279  ! half of the accumulated vertical mixing (eatr and ebtr) is applied in the call to tracer_column_fns.
280  ! For tracers whose source/sink terms need dt, this value is set to 1/2 dt_offline
281  ! 2) Half of the accumulated surface freshwater fluxes are applied
282  !! START ITERATION
283  ! 3) Accumulated mass fluxes are used to do horizontal transport. The number of iterations used in
284  ! advect_tracer is limited to 2 (e.g x->y->x->y). The remaining mass fluxes are stored for later use
285  ! and resulting layer thicknesses fed into the next step
286  ! 4) Tracers and the h-grid are regridded and remapped in a call to ALE. This allows for layers which might
287  ! 'vanish' because of horizontal mass transport to be 'reinflated'
288  ! 5) Check that transport is done if the remaining mass fluxes equals 0 or if the max number of iterations
289  ! has been reached
290  !! END ITERATION
291  ! 6) Repeat steps 1 and 2
292  ! 7) Force a remapping to the stored layer thicknesses that correspond to the snapshot of the online model
293  ! 8) Reset T/S and h to their stored snapshotted values to prevent model drift
294 
295  ! Copy over the horizontal mass fluxes from the total mass fluxes
296  do k=1,nz ; do j=jsd,jed ; do i=isdb,iedb
297  uhtr_sub(i,j,k) = uhtr(i,j,k)
298  enddo ; enddo ; enddo
299  do k=1,nz ; do j=jsdb,jedb ; do i=isd,ied
300  vhtr_sub(i,j,k) = vhtr(i,j,k)
301  enddo ; enddo ; enddo
302  do k=1,nz ; do j=js,je ; do i=is,ie

304  enddo ; enddo ; enddo
305

308  call uvchksum("[uv]htr_sub before transport", uhtr_sub, vhtr_sub, g%HI)
309  endif
310  tot_residual = remaining_transport_sum(cs, uhtr, vhtr)
311  if (cs%print_adv_offline) then
312  write(mesg,'(A,ES24.16)') "Main advection starting transport: ", tot_residual
313  call mom_mesg(mesg)
314  endif
315 
316  ! This loop does essentially a flux-limited, nonlinear advection scheme until all mass fluxes
317  ! are used. ALE is done after the horizontal advection.
318  do iter=1,cs%num_off_iter
319 
320  do k=1,nz ; do j=js,je ; do i=is,ie
321  h_vol(i,j,k) = h_new(i,j,k)*g%areaT(i,j)
322  h_pre(i,j,k) = h_new(i,j,k)
323  enddo ; enddo ; enddo
324 
325  if (cs%debug) then
326  call hchksum(h_vol,"h_vol before advect",g%HI)
327  call uvchksum("[uv]htr_sub before advect", uhtr_sub, vhtr_sub, g%HI)
328  write(debug_msg, '(A,I4.4)') 'Before advect ', iter
329  call mom_tracer_chkinv(debug_msg, g, h_pre, cs%tracer_reg%Tr, cs%tracer_reg%ntr)
330  endif
331 
332  call advect_tracer(h_pre, uhtr_sub, vhtr_sub, cs%OBC, cs%dt_offline, g, gv, &
333  cs%tracer_adv_CSp, cs%tracer_Reg, h_vol, max_iter_in=1, &
334  uhr_out=uhtr, vhr_out=vhtr, h_out=h_new, x_first_in=x_before_y)
335 
336  ! Switch the direction every iteration
337  x_before_y = .not. x_before_y
338 
339  ! Update the new layer thicknesses after one round of advection has happened
340  do k=1,nz ; do j=js,je ; do i=is,ie
341  h_new(i,j,k) = h_new(i,j,k)/g%areaT(i,j)
342  enddo ; enddo ; enddo
343 
344  if (modulo(iter,cs%off_ale_mod)==0) then
345  ! Do ALE remapping/regridding to allow for more advection to occur in the next iteration
346  call pass_var(h_new,g%Domain)
347  if (cs%debug) then
348  call hchksum(h_new,"h_new before ALE",g%HI)
349  write(debug_msg, '(A,I4.4)') 'Before ALE ', iter
350  call mom_tracer_chkinv(debug_msg, g, h_new, cs%tracer_reg%Tr, cs%tracer_reg%ntr)
351  endif
352  call cpu_clock_begin(id_clock_ale)
353  call ale_main_offline(g, gv, h_new, cs%tv, cs%tracer_Reg, cs%ALE_CSp, cs%dt_offline)
354  call cpu_clock_end(id_clock_ale)
355 
356  if (cs%debug) then
357  call hchksum(h_new,"h_new after ALE",g%HI)
358  write(debug_msg, '(A,I4.4)') 'After ALE ', iter
359  call mom_tracer_chkinv(debug_msg, g, h_new, cs%tracer_reg%Tr, cs%tracer_reg%ntr)
360  endif
361  endif
362 
363  do k=1,nz; do j=js,je ; do i=is,ie
364  uhtr_sub(i,j,k) = uhtr(i,j,k)
365  vhtr_sub(i,j,k) = vhtr(i,j,k)
366  enddo ; enddo ; enddo
367  call pass_var(h_new, g%Domain)
368  call pass_vector(uhtr_sub,vhtr_sub,g%Domain)
369 
370  ! Check for whether we've used up all the advection, or if we need to move on because
371  ! advection has stalled
372  tot_residual = remaining_transport_sum(cs, uhtr, vhtr)
373  if (cs%print_adv_offline) then
374  write(mesg,'(A,ES24.16)') "Main advection remaining transport: ", tot_residual
375  call mom_mesg(mesg)
376  endif
377  ! If all the mass transports have been used u, then quit
378  if (tot_residual == 0.0) then
379  write(mesg,*) "Converged after iteration ", iter
380  call mom_mesg(mesg)
381  converged = .true.
382  exit
383  endif
384  ! If advection has stalled or the remaining residual is less than a specified amount, quit
385  if ( (tot_residual == prev_tot_residual) .or. (tot_residual<cs%min_residual) ) then
386  converged = .false.
387  exit
388  endif
389 
390  prev_tot_residual = tot_residual
391 
392  enddo
393 
394  ! Make sure that uhtr and vhtr halos are updated
395  h_pre(:,:,:) = h_new(:,:,:)
396  call pass_vector(uhtr,vhtr,g%Domain)
397 
398  if (cs%debug) then
399  call hchksum(h_pre,"h after offline_advection_ale",g%HI)
400  call uvchksum("[uv]htr after offline_advection_ale", uhtr, vhtr, g%HI)
401  call mom_tracer_chkinv("After offline_advection_ale", g, h_pre, cs%tracer_reg%Tr, cs%tracer_reg%ntr)
402  endif
403 
404  call cpu_clock_end(cs%id_clock_offline_adv)

608 real function remaining_transport_sum(CS, uhtr, vhtr)

610  real, dimension(SZIB_(CS%G),SZJ_(CS%G),SZK_(CS%G)), intent(in ) :: uhtr !< Zonal mass transport

895  vhtr_sub(i,j,k) = vhtr(i,j,k)
896  enddo ; enddo ; enddo
897 
898 
899  ! Calculate 3d mass transports to be used in this iteration
900  call limit_mass_flux_3d(g, gv, uhtr_sub, vhtr_sub, eatr_sub, ebtr_sub, h_pre)
901 
902  if (z_first) then
903  ! First do vertical advection
904  call update_h_vertical_flux(g, gv, eatr_sub, ebtr_sub, h_pre, h_new)
905  call call_tracer_column_fns(h_pre, h_new, eatr_sub, ebtr_sub, &
906  fluxes, cs%mld, dt_iter, g, gv, cs%tv, cs%optics, cs%tracer_flow_CSp, cs%debug)
907  ! We are now done with the vertical mass transports, so now h_new is h_sub
908  do k = 1, nz ; do j=js-1,je+1 ; do i=is-1,ie+1
909  h_pre(i,j,k) = h_new(i,j,k)
910  enddo ; enddo ; enddo
911  call pass_var(h_pre,g%Domain)
912 
913  ! Second zonal and meridional advection
914  call update_h_horizontal_flux(g, gv, uhtr_sub, vhtr_sub, h_pre, h_new)
915  do k = 1, nz ; do i = is-1, ie+1 ; do j=js-1, je+1
916  h_vol(i,j,k) = h_pre(i,j,k)*g%areaT(i,j)
917  enddo ; enddo ; enddo
918  call advect_tracer(h_pre, uhtr_sub, vhtr_sub, cs%OBC, dt_iter, g, gv, &
919  cs%tracer_adv_CSp, cs%tracer_Reg, h_vol, max_iter_in=30, x_first_in=x_before_y)
920 
921  ! Done with horizontal so now h_pre should be h_new
922  do k = 1, nz ; do i=is-1,ie+1 ; do j=js-1,je+1
923  h_pre(i,j,k) = h_new(i,j,k)
924  enddo ; enddo ; enddo
925 
926  endif
927 
928  if (.not. z_first) then
929 
930  ! First zonal and meridional advection
931  call update_h_horizontal_flux(g, gv, uhtr_sub, vhtr_sub, h_pre, h_new)
932  do k = 1, nz ; do i = is-1, ie+1 ; do j=js-1, je+1
933  h_vol(i,j,k) = h_pre(i,j,k)*g%areaT(i,j)
934  enddo ; enddo ; enddo
935  call advect_tracer(h_pre, uhtr_sub, vhtr_sub, cs%OBC, dt_iter, g, gv, &
936  cs%tracer_adv_CSp, cs%tracer_Reg, h_vol, max_iter_in=30, x_first_in=x_before_y)
937 
938  ! Done with horizontal so now h_pre should be h_new
939  do k = 1, nz ; do i=is-1,ie+1 ; do j=js-1,je+1
940  h_pre(i,j,k) = h_new(i,j,k)
941  enddo ; enddo ; enddo
942 
943  ! Second vertical advection
944  call update_h_vertical_flux(g, gv, eatr_sub, ebtr_sub, h_pre, h_new)
945  call call_tracer_column_fns(h_pre, h_new, eatr_sub, ebtr_sub, &
946  fluxes, cs%mld, dt_iter, g, gv, cs%tv, cs%optics, cs%tracer_flow_CSp, cs%debug)
947  ! We are now done with the vertical mass transports, so now h_new is h_sub
948  do k = 1, nz ; do i=is-1,ie+1 ; do j=js-1,je+1
949  h_pre(i,j,k) = h_new(i,j,k)
950  enddo ; enddo ; enddo
951 
952  endif
953 
954  ! Update remaining transports
955  do k = 1, nz ; do j=js-1,je+1 ; do i=is-1,ie+1
956  eatr(i,j,k) = eatr(i,j,k) - eatr_sub(i,j,k)
957  ebtr(i,j,k) = ebtr(i,j,k) - ebtr_sub(i,j,k)
958  enddo ; enddo ; enddo
959 
960  do k = 1, nz ; do j=js-1,je+1 ; do i=is-2,ie+1
961  uhtr(i,j,k) = uhtr(i,j,k) - uhtr_sub(i,j,k)
962  enddo ; enddo ; enddo
963 
964  do k = 1, nz ; do j=js-2,je+1 ; do i=is-1,ie+1
965  vhtr(i,j,k) = vhtr(i,j,k) - vhtr_sub(i,j,k)
966  enddo ; enddo ; enddo
967 
968  call pass_var(eatr,g%Domain)
969  call pass_var(ebtr,g%Domain)
970  call pass_var(h_pre,g%Domain)
971  call pass_vector(uhtr,vhtr,g%Domain)
972  !
973  ! Calculate how close we are to converging by summing the remaining fluxes at each point
974  sum_abs_fluxes = 0.0
975  sum_u = 0.0
976  sum_v = 0.0
977  do k=1,nz; do j=js,je; do i=is,ie
978  sum_u = sum_u + abs(uhtr(i-1,j,k))+abs(uhtr(i,j,k))
979  sum_v = sum_v + abs(vhtr(i,j-1,k))+abs(vhtr(i,j,k))
980  sum_abs_fluxes = sum_abs_fluxes + abs(eatr(i,j,k)) + abs(ebtr(i,j,k)) + abs(uhtr(i-1,j,k)) + &
981  abs(uhtr(i,j,k)) + abs(vhtr(i,j-1,k)) + abs(vhtr(i,j,k))
982  enddo ; enddo ; enddo
983  call sum_across_pes(sum_abs_fluxes)
984 
985  write(mesg,*) "offline_advection_layer: Remaining u-flux, v-flux:", sum_u, sum_v
986  call mom_mesg(mesg)
987  if (sum_abs_fluxes==0) then
988  write(mesg,*) 'offline_advection_layer: Converged after iteration', iter
989  call mom_mesg(mesg)
990  exit
991  endif
992 
993  ! Switch order of Strang split every iteration
994  z_first = .not. z_first
995  x_before_y = .not. x_before_y
996  enddo

1506 !! \section offline_overview Offline Tracer Transport in MOM6
1507 !! 'Offline tracer modeling' uses physical fields (e.g. mass transports and layer thicknesses) saved
1508 !! from a previous integration of the physical model to transport passive tracers. These fields are
1509 !! accumulated or averaged over a period of time (in this test case, 1 day) and used to integrate
1510 !! portions of the MOM6 code base that handle the 3d advection and diffusion of passive tracers.
1511 !!
1512 !! The distribution of tracers in the ocean modeled offline should not be expected to match an online
1513 !! simulation. Accumulating transports over more than one online model timestep implicitly assumes
1514 !! homogeneity over that time period and essentially aliases over processes that occur with higher
1515 !! frequency. For example, consider the case of a surface boundary layer with a strong diurnal cycle.
1516 !! An offline simulation with a 1 day timestep, captures the net transport into or out of that layer,
1517 !! but not the exact cycling. This effective aliasing may also complicate online model configurations
1518 !! which strongly-eddying regions. In this case, the offline model timestep must be limited to some
1519 !! fraction of the eddy correlation timescale. Lastly, the nonlinear advection scheme which applies
1520 !! limited mass-transports over a sequence of iterations means that tracers are not transported along
1521 !! exactly the same path as they are in the online model.
1522 !!
1523 !! This capability has currently targeted the Baltic_ALE_z test case, though some work has also been
1524 !! done with the OM4 1/2 degree configuration. Work is ongoing to develop recommendations and best
1525 !! practices for investigators seeking to use MOM6 for offline tracer modeling.
1526 !!
1527 !! \section offline_technical Implementation of offline routine in MOM6
1528 !!
1529 !! The subroutine step_tracers that coordinates this can be found in MOM.F90 and is only called
1530 !! using the solo ocean driver. This is to avoid issues with coupling to other climate components
1531 !! that may be relying on fluxes from the ocean to be coupled more often than the offline time step.
1532 !! Other routines related to offline tracer modeling can be found in tracers/MOM_offline_control.F90
1533 !!
1534 !! As can also be seen in the comments for the step_tracers subroutine, an offline time step
1535 !! comprises the following steps:
1536 !! -# Using the layer thicknesses and tracer concentrations from the previous timestep,
1537 !! half of the accumulated vertical mixing (eatr and ebtr) is applied in the call to
1538 !! tracer_column_fns.
1539 !! For tracers whose source/sink terms need dt, this value is set to 1/2 dt_offline
1540 !! -# Half of the accumulated surface freshwater fluxes are applied
1541 !! START ITERATION
1542 !! -# Accumulated mass fluxes are used to do horizontal transport. The number of iterations
1543 !! used in advect_tracer is limited to 2 (e.g x->y->x->y). The remaining mass fluxes are
1544 !! stored for later use and resulting layer thicknesses fed into the next step
1545 !! -# Tracers and the h-grid are regridded and remapped in a call to ALE. This allows for
1546 !! layers which might 'vanish' because of horizontal mass transport to be 'reinflated'
1547 !! and essentially allows for the vertical transport of tracers
1548 !! -# Check that transport is done if the remaining mass fluxes equals 0 or if the max
1549 !! number of iterations has been reached
1550 !! END ITERATION
1551 !! -# Repeat steps 1 and 2
1552 !! -# Redistribute any residual mass fluxes that remain after the advection iterations
1553 !! in a barotropic manner, progressively upward through the water column.
1554 !! -# Force a remapping to the stored layer thicknesses that correspond to the snapshot of
1555 !! the online model at the end of an accumulation interval
1556 !! -# Reset T/S and h to their stored snapshotted values to prevent model drift
1557 !!
1558 !! \section offline_evaluation Evaluating the utility of an offline tracer model
1559 !! How well an offline tracer model can be used as an alternative to integrating tracers online
1560 !! with the prognostic model must be evaluated for each application. This efficacy may be related
1561 !! to the native coordinate of the online model, to the length of the offline timestep, and to the
1562 !! behavior of the tracer itself.
1563 !!
1564 !! A framework for formally regression testing the offline capability still needs to be developed.
1565 !! However, as a simple way of testing whether the offline model is nominally behaving as expected,
1566 !! the total inventory of the advection test tracers (tr1, tr2, etc.) should be conserved between
1567 !! time steps except for the last 4 decimal places. As a general guideline, an offline timestep of
1568 !! 5 days or less.
1569 !!
1570 !! \section offline_parameters Runtime parameters for offline tracers
1571 !! - OFFLINEDIR: Input directory where the offline fields can be found
1572 !! - OFF_SUM_FILE: Filename where the accumulated fields can be found (e.g. horizontal mass transports)
1573 !! - OFF_SNAP_FILE: Filename where snapshot fields can be found (e.g. end of timestep layer thickness)
1574 !! - START_INDEX: Which timelevel of the input files to read first
1575 !! - NUMTIME: How many timelevels to read before 'looping' back to 1
1576 !! - FIELDS_ARE_OFFSET: True if the time-averaged fields and snapshot fields are offset by one
1577 !! time level, probably not needed
1578 !! -NUM_OFF_ITER: Maximum number of iterations to do for the nonlinear advection scheme
1579 !! -REDISTRIBUTE_METHOD: Redistributes any remaining horizontal fluxes throughout the rest of water column.
1580 !! Options are 'barotropic' which "evenly distributes flux throughout the entire water
1581 !! column,'upwards' which adds the maximum of the remaining flux in each layer above,
1582 !! and 'none' which does no redistribution"
