48 !> Bouyancy forcing for the boundary-forced-basin (BFB) configuration
49 subroutine bfb_buoyancy_forcing(state, fluxes, day, dt, G, CS)
50  type(surface), intent(inout) :: state !< A structure containing fields that
51  !! describe the surface state of the ocean.
52  type(forcing), intent(inout) :: fluxes !< A structure containing pointers to any
53  !! possible forcing fields. Unused fields
54  !! have NULL ptrs.
55  type(time_type), intent(in) :: day !< Time of the fluxes.
56  real, intent(in) :: dt !< The amount of time over which
57  !! the fluxes apply [s]
58  type(ocean_grid_type), intent(in) :: g !< The ocean's grid structure
59  type(bfb_surface_forcing_cs), pointer :: cs !< A pointer to the control structure
60  !! returned by a previous call to
61  !! BFB_surface_forcing_init.
62  ! Local variables
63  real :: temp_restore ! The temperature that is being restored toward [degC].
64  real :: salin_restore ! The salinity that is being restored toward [ppt].
65  real :: density_restore ! The potential density that is being restored
66  ! toward [kg m-3].
67  real :: rhoxcp ! The mean density times the heat capacity [J m-3 degC-1].
68  real :: buoy_rest_const ! A constant relating density anomalies to the
69  ! restoring buoyancy flux [m5 s-3 kg-1].
70  integer :: i, j, is, ie, js, je
71  integer :: isd, ied, jsd, jed
72 
73  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec
74  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
75 
76  ! Allocate and zero out the forcing arrays, as necessary. This portion is
77  ! usually not changed.
78  if (cs%use_temperature) then
79  call safe_alloc_ptr(fluxes%evap, isd, ied, jsd, jed)
80  call safe_alloc_ptr(fluxes%lprec, isd, ied, jsd, jed)
81  call safe_alloc_ptr(fluxes%fprec, isd, ied, jsd, jed)
82  call safe_alloc_ptr(fluxes%lrunoff, isd, ied, jsd, jed)
83  call safe_alloc_ptr(fluxes%frunoff, isd, ied, jsd, jed)
84  call safe_alloc_ptr(fluxes%vprec, isd, ied, jsd, jed)
85 
86  call safe_alloc_ptr(fluxes%sw, isd, ied, jsd, jed)
87  call safe_alloc_ptr(fluxes%lw, isd, ied, jsd, jed)
88  call safe_alloc_ptr(fluxes%latent, isd, ied, jsd, jed)
89  call safe_alloc_ptr(fluxes%sens, isd, ied, jsd, jed)
90  else ! This is the buoyancy only mode.
91  call safe_alloc_ptr(fluxes%buoy, isd, ied, jsd, jed)
92  endif
93 
94  if ( cs%use_temperature ) then
95  ! Set whichever fluxes are to be used here. Any fluxes that
96  ! are always zero do not need to be changed here.
97  do j=js,je ; do i=is,ie
98  ! Fluxes of fresh water through the surface are in units of [kg m-2 s-1]
99  ! and are positive downward - i.e. evaporation should be negative.
100  fluxes%evap(i,j) = -0.0 * g%mask2dT(i,j)
101  fluxes%lprec(i,j) = 0.0 * g%mask2dT(i,j)
102 
103  ! vprec will be set later, if it is needed for salinity restoring.
104  fluxes%vprec(i,j) = 0.0
105 
106  ! Heat fluxes are in units of [W m-2] and are positive into the ocean.
107  fluxes%lw(i,j) = 0.0 * g%mask2dT(i,j)
108  fluxes%latent(i,j) = 0.0 * g%mask2dT(i,j)
109  fluxes%sens(i,j) = 0.0 * g%mask2dT(i,j)
110  fluxes%sw(i,j) = 0.0 * g%mask2dT(i,j)
111  enddo ; enddo
112  else ! This is the buoyancy only mode.
113  do j=js,je ; do i=is,ie
114  ! fluxes%buoy is the buoyancy flux into the ocean [m2 s-3]. A positive
115  ! buoyancy flux is of the same sign as heating the ocean.
116  fluxes%buoy(i,j) = 0.0 * g%mask2dT(i,j)
117  enddo ; enddo
118  endif
119 
120  if (cs%restorebuoy) then
121  if (cs%use_temperature) then
122  call safe_alloc_ptr(fluxes%heat_added, isd, ied, jsd, jed)
123  ! When modifying the code, comment out this error message. It is here
124  ! so that the original (unmodified) version is not accidentally used.
125  call mom_error(fatal, "User_buoyancy_surface_forcing: " // &
126  "Temperature and salinity restoring used without modification." )
127 
128  rhoxcp = cs%Rho0 * fluxes%C_p
129  do j=js,je ; do i=is,ie
130  ! Set Temp_restore and Salin_restore to the temperature (in degC) and
131  ! salinity (in ppt) that are being restored toward.
132  temp_restore = 0.0
133  salin_restore = 0.0
134 
135  fluxes%heat_added(i,j) = (g%mask2dT(i,j) * (rhoxcp * cs%Flux_const)) * &
136  (temp_restore - state%SST(i,j))
137  fluxes%vprec(i,j) = - (g%mask2dT(i,j) * (cs%Rho0*cs%Flux_const)) * &
138  ((salin_restore - state%SSS(i,j)) / &
139  (0.5 * (salin_restore + state%SSS(i,j))))
140  enddo ; enddo
141  else
142  ! When modifying the code, comment out this error message. It is here
143  ! so that the original (unmodified) version is not accidentally used.
144  ! call MOM_error(FATAL, "User_buoyancy_surface_forcing: " // &
145  ! "Buoyancy restoring used without modification." )
146 
147  ! The -1 is because density has the opposite sign to buoyancy.
148  buoy_rest_const = -1.0 * (cs%G_Earth * cs%Flux_const) / cs%Rho0
149  temp_restore = 0.0
150  do j=js,je ; do i=is,ie
151  ! Set density_restore to an expression for the surface potential
152  ! density [kg m-3] that is being restored toward.
153  if (g%geoLatT(i,j) < cs%lfrslat) then
154  temp_restore = cs%SST_s
155  elseif (g%geoLatT(i,j) > cs%lfrnlat) then
156  temp_restore = cs%SST_n
157  else
158  temp_restore = (cs%SST_s - cs%SST_n)/(cs%lfrslat - cs%lfrnlat) * &
159  (g%geoLatT(i,j) - cs%lfrslat) + cs%SST_s
160  endif
161 
162  density_restore = temp_restore*cs%drho_dt + cs%Rho0
163 
164  fluxes%buoy(i,j) = g%mask2dT(i,j) * buoy_rest_const * &
165  (density_restore - state%sfc_density(i,j))
166  enddo ; enddo
167  endif
168  endif ! end RESTOREBUOY
169 
170 end subroutine bfb_buoyancy_forcing
