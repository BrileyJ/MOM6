61 !> Sets the model-state dependent internal tide energy sources.
62 subroutine set_int_tide_input(u, v, h, tv, fluxes, itide, dt, G, GV, US, CS)
63  type(ocean_grid_type), intent(in) :: g !< The ocean's grid structure
64  type(verticalgrid_type), intent(in) :: gv !< The ocean's vertical grid structure
65  type(unit_scale_type), intent(in) :: us !< A dimensional unit scaling type
66  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(in) :: u !< The zonal velocity [m s-1]
67  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(in) :: v !< The meridional velocity [m s-1]
68  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h !< Layer thicknesses [H ~> m or kg m-2]
69  type(thermo_var_ptrs), intent(in) :: tv !< A structure containing pointers to the
70  !! thermodynamic fields
71  type(forcing), intent(in) :: fluxes !< A structure of thermodynamic surface fluxes
72  type(int_tide_input_type), intent(inout) :: itide !< A structure containing fields related
73  !! to the internal tide sources.
74  real, intent(in) :: dt !< The time increment [s].
75  type(int_tide_input_cs), pointer :: cs !< This module's control structure.
76  ! Local variables
77  real, dimension(SZI_(G),SZJ_(G)) :: &
78  n2_bot ! The bottom squared buoyancy frequency [s-2].
79 
80  real, dimension(SZI_(G),SZJ_(G),SZK_(G)) :: &
81  t_f, s_f ! The temperature and salinity in [degC] and [ppt] with the values in
82  ! the massless layers filled vertically by diffusion.
83  logical :: use_eos ! If true, density is calculated from T & S using an
84  ! equation of state.
85  integer :: i, j, k, is, ie, js, je, nz
86  integer :: isd, ied, jsd, jed
87 
88  real :: kappa_fill ! diffusivity used to fill massless layers
89  real :: dt_fill ! timestep used to fill massless layers
90 
91  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
92  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
93 
94  if (.not.associated(cs)) call mom_error(fatal,"set_diffusivity: "//&
95  "Module must be initialized before it is used.")
96 
97  kappa_fill = 1.e-3*us%m_to_Z**2 !### Dimensional constant [m2 s-1].
98  dt_fill = 7200. !### Dimensionalconstant [s].
99 
100  use_eos = associated(tv%eqn_of_state)
101 
102  ! Smooth the properties through massless layers.
103  if (use_eos) then
104  call vert_fill_ts(h, tv%T, tv%S, kappa_fill, dt_fill, t_f, s_f, g, gv)
105  endif
106 
107  call find_n2_bottom(h, tv, t_f, s_f, itide%h2, fluxes, g, gv, us, n2_bot)
108 
109 !$OMP parallel do default(none) shared(is,ie,js,je,G,itide,N2_bot,CS)
110  do j=js,je ; do i=is,ie
111  itide%Nb(i,j) = g%mask2dT(i,j) * sqrt(n2_bot(i,j))
112  itide%TKE_itidal_input(i,j) = min(cs%TKE_itidal_coef(i,j)*itide%Nb(i,j),cs%TKE_itide_max)
113  enddo ; enddo
114 
115  if (cs%debug) then
116  call hchksum(n2_bot,"N2_bot",g%HI,haloshift=0)
117  call hchksum(itide%TKE_itidal_input,"TKE_itidal_input",g%HI,haloshift=0)
118  endif
119 
120  if (cs%id_TKE_itidal > 0) call post_data(cs%id_TKE_itidal, itide%TKE_itidal_input, cs%diag)
121  if (cs%id_Nb > 0) call post_data(cs%id_Nb, itide%Nb, cs%diag)
122  if (cs%id_N2_bot > 0 ) call post_data(cs%id_N2_bot,n2_bot,cs%diag)
123 
124 end subroutine set_int_tide_input

126 !> Estimates the near-bottom buoyancy frequency (N^2).
127 subroutine find_n2_bottom(h, tv, T_f, S_f, h2, fluxes, G, GV, US, N2_bot)
128  type(ocean_grid_type), intent(in) :: G !< The ocean's grid structure
129  type(verticalgrid_type), intent(in) :: GV !< The ocean's vertical grid structure
130  type(unit_scale_type), intent(in) :: US !< A dimensional unit scaling type
131  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h !< Layer thicknesses [H ~> m or kg m-2]
132  type(thermo_var_ptrs), intent(in) :: tv !< A structure containing pointers to the
133  !! thermodynamic fields
134  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: T_f !< Temperature after vertical filtering to
135  !! smooth out the values in thin layers [degC].
136  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: S_f !< Salinity after vertical filtering to
137  !! smooth out the values in thin layers [ppt].
138  real, dimension(SZI_(G),SZJ_(G)), intent(in) :: h2 !< Bottom topographic roughness [Z2 ~> m2].
139  type(forcing), intent(in) :: fluxes !< A structure of thermodynamic surface fluxes
140  type(int_tide_input_cs), pointer :: CS !< This module's control structure.
141  real, dimension(SZI_(G),SZJ_(G)), intent(out) :: N2_bot !< The squared buoyancy freqency at the
142  !! ocean bottom [s-2].
143  ! Local variables
144  real, dimension(SZI_(G),SZK_(G)+1) :: &
145  dRho_int ! The unfiltered density differences across interfaces.
146  real, dimension(SZI_(G)) :: &
147  pres, & ! The pressure at each interface [Pa].
148  Temp_int, & ! The temperature at each interface [degC].
149  Salin_int, & ! The salinity at each interface [ppt].
150  drho_bot, &
151  h_amp, & ! The amplitude of topographic roughness [Z ~> m].
152  hb, & ! The depth below a layer [Z ~> m].
153  z_from_bot, & ! The height of a layer center above the bottom [Z ~> m].
154  dRho_dT, & ! The partial derivatives of density with temperature and
155  dRho_dS ! salinity [kg m-3 degC-1] and [kg m-3 ppt-1].
156 
157  real :: dz_int ! The thickness associated with an interface [Z ~> m].
158  real :: G_Rho0 ! The gravitation acceleration divided by the Boussinesq
159  ! density [Z m3 s-2 kg-1 ~> m4 s-2 kg-1].
160  logical :: do_i(SZI_(G)), do_any
161  integer :: i, j, k, is, ie, js, je, nz
162  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
163  g_rho0 = (gv%g_Earth*us%m_to_Z**2) / gv%Rho0
164 
165  ! Find the (limited) density jump across each interface.
166  do i=is,ie
167  drho_int(i,1) = 0.0 ; drho_int(i,nz+1) = 0.0
168  enddo
169 !$OMP parallel do default(none) shared(is,ie,js,je,nz,tv,fluxes,G,GV,h,T_f,S_f, &
170 !$OMP h2,N2_bot,G_Rho0) &
171 !$OMP private(pres,Temp_Int,Salin_Int,dRho_dT,dRho_dS, &
172 !$OMP hb,dRho_bot,z_from_bot,do_i,h_amp, &
173 !$OMP do_any,dz_int) &
174 !$OMP firstprivate(dRho_int)
175  do j=js,je
176  if (associated(tv%eqn_of_state)) then
177  if (associated(fluxes%p_surf)) then
178  do i=is,ie ; pres(i) = fluxes%p_surf(i,j) ; enddo
179  else
180  do i=is,ie ; pres(i) = 0.0 ; enddo
181  endif
182  do k=2,nz
183  do i=is,ie
184  pres(i) = pres(i) + gv%H_to_Pa*h(i,j,k-1)
185  temp_int(i) = 0.5 * (t_f(i,j,k) + t_f(i,j,k-1))
186  salin_int(i) = 0.5 * (s_f(i,j,k) + s_f(i,j,k-1))
187  enddo
188  call calculate_density_derivs(temp_int, salin_int, pres, &
189  drho_dt(:), drho_ds(:), is, ie-is+1, tv%eqn_of_state)
190  do i=is,ie
191  drho_int(i,k) = max(drho_dt(i)*(t_f(i,j,k) - t_f(i,j,k-1)) + &
192  drho_ds(i)*(s_f(i,j,k) - s_f(i,j,k-1)), 0.0)
193  enddo
194  enddo
195  else
196  do k=2,nz ; do i=is,ie
197  drho_int(i,k) = gv%Rlay(k) - gv%Rlay(k-1)
198  enddo ; enddo
199  endif
200 
201  ! Find the bottom boundary layer stratification.
202  do i=is,ie
203  hb(i) = 0.0 ; drho_bot(i) = 0.0
204  z_from_bot(i) = 0.5*gv%H_to_Z*h(i,j,nz)
205  do_i(i) = (g%mask2dT(i,j) > 0.5)
206  h_amp(i) = sqrt(h2(i,j))
207  enddo
208 
209  do k=nz,2,-1
210  do_any = .false.
211  do i=is,ie ; if (do_i(i)) then
212  dz_int = 0.5*gv%H_to_Z*(h(i,j,k) + h(i,j,k-1))
213  z_from_bot(i) = z_from_bot(i) + dz_int ! middle of the layer above
214 
215  hb(i) = hb(i) + dz_int
216  drho_bot(i) = drho_bot(i) + drho_int(i,k)
217 
218  if (z_from_bot(i) > h_amp(i)) then
219  if (k>2) then
220  ! Always include at least one full layer.
221  hb(i) = hb(i) + 0.5*gv%H_to_Z*(h(i,j,k-1) + h(i,j,k-2))
222  drho_bot(i) = drho_bot(i) + drho_int(i,k-1)
223  endif
224  do_i(i) = .false.
225  else
226  do_any = .true.
227  endif
228  endif ; enddo
229  if (.not.do_any) exit
230  enddo
231 
232  do i=is,ie
233  if (hb(i) > 0.0) then
234  n2_bot(i,j) = (g_rho0 * drho_bot(i)) / hb(i)
235  else ; n2_bot(i,j) = 0.0 ; endif
236  enddo
237  enddo
238 
239 end subroutine find_n2_bottom
