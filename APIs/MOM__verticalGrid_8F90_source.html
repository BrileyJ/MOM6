18 ! A note on unit descriptions in comments: MOM6 uses units that can be rescaled for dimensional
19 ! consistency testing. These are noted in comments with units like Z, H, L, and T, along with
20 ! their mks counterparts with notation like "a velocity [Z T-1 ~> m s-1]". If the units
21 ! vary with the Boussinesq approximation, the Boussinesq variant is given first.

69 !> Allocates and initializes the ocean model vertical grid structure.
70 subroutine verticalgridinit( param_file, GV, US )
71  type(param_file_type), intent(in) :: param_file !< Parameter file handle/type
72  type(verticalgrid_type), pointer :: gv !< The container for vertical grid data
73  type(unit_scale_type), intent(in) :: us !< A dimensional unit scaling type
74  ! This routine initializes the verticalGrid_type structure (GV).
75  ! All memory is allocated but not necessarily set to meaningful values until later.
76 
77  ! Local variables
78  integer :: nk, h_power
79  real :: h_rescale_factor
80  ! This include declares and sets the variable "version".
81 # include "version_variable.h"
82  character(len=16) :: mdl = 'MOM_verticalGrid'
83 
84  if (associated(gv)) call mom_error(fatal, &
85  'verticalGridInit: called with an associated GV pointer.')
86  allocate(gv)
87 
88  ! Read all relevant parameters and write them to the model log.
89  call log_version(param_file, mdl, version, &
90  "Parameters providing information about the vertical grid.")
91  call get_param(param_file, mdl, "G_EARTH", gv%g_Earth, &
92  "The gravitational acceleration of the Earth.", &
93  units="m s-2", default = 9.80)
94  call get_param(param_file, mdl, "RHO_0", gv%Rho0, &
95  "The mean ocean density used with BOUSSINESQ true to "//&
96  "calculate accelerations and the mass for conservation "//&
97  "properties, or with BOUSSINSEQ false to convert some "//&
98  "parameters from vertical units of m to kg m-2.", &
99  units="kg m-3", default=1035.0)
100  call get_param(param_file, mdl, "BOUSSINESQ", gv%Boussinesq, &
101  "If true, make the Boussinesq approximation.", default=.true.)
102  call get_param(param_file, mdl, "ANGSTROM", gv%Angstrom_m, &
103  "The minimum layer thickness, usually one-Angstrom.", &
104  units="m", default=1.0e-10)
105  call get_param(param_file, mdl, "H_RESCALE_POWER", h_power, &
106  "An integer power of 2 that is used to rescale the model's "//&
107  "intenal units of thickness. Valid values range from -300 to 300.", &
108  units="nondim", default=0, debuggingparam=.true.)
109  if (abs(h_power) > 300) call mom_error(fatal, "verticalGridInit: "//&
110  "H_RESCALE_POWER is outside of the valid range of -300 to 300.")
111  h_rescale_factor = 1.0
112  if (h_power /= 0) h_rescale_factor = 2.0**h_power
113  if (.not.gv%Boussinesq) then
114  call get_param(param_file, mdl, "H_TO_KG_M2", gv%H_to_kg_m2,&
115  "A constant that translates thicknesses from the model's "//&
116  "internal units of thickness to kg m-2.", units="kg m-2 H-1", &
117  default=1.0)
118  gv%H_to_kg_m2 = gv%H_to_kg_m2 * h_rescale_factor
119  else
120  call get_param(param_file, mdl, "H_TO_M", gv%H_to_m, &
121  "A constant that translates the model's internal "//&
122  "units of thickness into m.", units="m H-1", default=1.0)
123  gv%H_to_m = gv%H_to_m * h_rescale_factor
124  endif
125  gv%g_Earth = gv%g_Earth * us%Z_to_m
126 #ifdef STATIC_MEMORY_
127  ! Here NK_ is a macro, while nk is a variable.
128  call get_param(param_file, mdl, "NK", nk, &
129  "The number of model layers.", units="nondim", &
130  static_value=nk_)
131  if (nk /= nk_) call mom_error(fatal, "verticalGridInit: " // &
132  "Mismatched number of layers NK_ between MOM_memory.h and param_file")
133 
134 #else
135  call get_param(param_file, mdl, "NK", nk, &
136  "The number of model layers.", units="nondim", fail_if_missing=.true.)
137 #endif
138  gv%ke = nk
139 
140  if (gv%Boussinesq) then
141  gv%H_to_kg_m2 = gv%Rho0 * gv%H_to_m
142  gv%kg_m2_to_H = 1.0 / gv%H_to_kg_m2
143  gv%m_to_H = 1.0 / gv%H_to_m
144  gv%Angstrom_H = gv%m_to_H * gv%Angstrom_m
145  else
146  gv%kg_m2_to_H = 1.0 / gv%H_to_kg_m2
147  gv%m_to_H = gv%Rho0 * gv%kg_m2_to_H
148  gv%H_to_m = gv%H_to_kg_m2 / gv%Rho0
149  gv%Angstrom_H = gv%Angstrom_m*1000.0*gv%kg_m2_to_H
150  endif
151  gv%H_subroundoff = 1e-20 * max(gv%Angstrom_H,gv%m_to_H*1e-17)
152  gv%H_to_Pa = (gv%g_Earth*us%m_to_Z) * gv%H_to_kg_m2
153 
154  gv%H_to_Z = gv%H_to_m * us%m_to_Z
155  gv%Z_to_H = us%Z_to_m * gv%m_to_H
156  gv%Angstrom_Z = us%m_to_Z * gv%Angstrom_m
157 
158 ! Log derivative values.
159  call log_param(param_file, mdl, "M to THICKNESS", gv%m_to_H*h_rescale_factor)
160  call log_param(param_file, mdl, "M to THICKNESS rescaled by 2^-n", gv%m_to_H)
161  call log_param(param_file, mdl, "THICKNESS to M rescaled by 2^n", gv%H_to_m)
162 
163  allocate( gv%sInterface(nk+1) )
164  allocate( gv%sLayer(nk) )
165  allocate( gv%g_prime(nk+1) ) ; gv%g_prime(:) = 0.0
166  ! The extent of Rlay should be changed to nk?
167  allocate( gv%Rlay(nk+1) ) ; gv%Rlay(:) = 0.0
168 
169 end subroutine verticalgridinit

208 !> Returns the model's tracer flux units.
209 function get_tr_flux_units(GV, tr_units, tr_vol_conc_units, tr_mass_conc_units)
210  character(len=48) :: get_tr_flux_units !< The model's flux units
211  !! for a tracer.
212  type(verticalgrid_type), intent(in) :: gv !< The ocean's vertical
213  !! grid structure.
214  character(len=*), optional, intent(in) :: tr_units !< Units for a tracer, for example
215  !! Celsius or PSU.
216  character(len=*), optional, intent(in) :: tr_vol_conc_units !< The concentration units per unit
217  !! volume, for example if the units are
218  !! umol m-3, tr_vol_conc_units would
219  !! be umol.
220  character(len=*), optional, intent(in) :: tr_mass_conc_units !< The concentration units per unit
221  !! mass of sea water, for example if
222  !! the units are mol kg-1,
223  !! tr_vol_conc_units would be mol.
224 
225  ! This subroutine returns the appropriate units for thicknesses and fluxes,
226  ! depending on whether the model is Boussinesq or not and the scaling for
227  ! the vertical thickness.
228  integer :: cnt
229 
230  cnt = 0
231  if (present(tr_units)) cnt = cnt+1
232  if (present(tr_vol_conc_units)) cnt = cnt+1
233  if (present(tr_mass_conc_units)) cnt = cnt+1
234 
235  if (cnt == 0) call mom_error(fatal, "get_tr_flux_units: One of the three "//&
236  "arguments tr_units, tr_vol_conc_units, or tr_mass_conc_units "//&
237  "must be present.")
238  if (cnt > 1) call mom_error(fatal, "get_tr_flux_units: Only one of "//&
239  "tr_units, tr_vol_conc_units, and tr_mass_conc_units may be present.")
240  if (present(tr_units)) then
241  if (gv%Boussinesq) then
242  get_tr_flux_units = trim(tr_units)//" m3 s-1"
243  else
244  get_tr_flux_units = trim(tr_units)//" kg s-1"
245  endif
246  endif
247  if (present(tr_vol_conc_units)) then
248  if (gv%Boussinesq) then
249  get_tr_flux_units = trim(tr_vol_conc_units)//" s-1"
250  else
251  get_tr_flux_units = trim(tr_vol_conc_units)//" m-3 kg s-1"
252  endif
253  endif
254  if (present(tr_mass_conc_units)) then
255  if (gv%Boussinesq) then
256  get_tr_flux_units = trim(tr_mass_conc_units)//" kg-1 m3 s-1"
257  else
258  get_tr_flux_units = trim(tr_mass_conc_units)//" s-1"
259  endif
260  endif
261 
262 end function get_tr_flux_units

264 !> This sets the coordinate data for the "layer mode" of the isopycnal model.
265 subroutine setverticalgridaxes( Rlay, GV )
266  type(verticalgrid_type), intent(inout) :: gv !< The container for vertical grid data
267  real, dimension(GV%ke), intent(in) :: rlay !< The layer target density
268  ! Local variables
269  integer :: k, nk
270 
271  nk = gv%ke
272 
273  gv%zAxisLongName = 'Target Potential Density'
274  gv%zAxisUnits = 'kg m-3'
275  do k=1,nk ; gv%sLayer(k) = rlay(k) ; enddo
276  if (nk > 1) then
277  gv%sInterface(1) = 1.5*rlay(1) - 0.5*rlay(2)
278  do k=2,nk ; gv%sInterface(k) = 0.5*( rlay(k-1) + rlay(k) ) ; enddo
279  gv%sInterface(nk+1) = 1.5*rlay(nk) - 0.5*rlay(nk-1)
280  else
281  gv%sInterface(1) = 0.0 ; gv%sInterface(nk+1) = 2.0*rlay(nk)
282  endif
283 
284 end subroutine setverticalgridaxes
