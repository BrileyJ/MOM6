1296 !> Tridiagonal solver for streamfunction at interfaces
1297 subroutine streamfn_solver(nk, c2_h, hN2, sfn)
1298  integer, intent(in) :: nk !< Number of layers
1299  real, dimension(nk), intent(in) :: c2_h !< Wave speed squared over thickness in layers [m s-2]
1300  real, dimension(nk+1), intent(in) :: hN2 !< Thickness times N2 at interfaces [m s-2]
1301  real, dimension(nk+1), intent(inout) :: sfn !< Streamfunction [Z m2 s-1 ~> m3 s-1] or arbitrary units
1302  !! On entry, equals diffusivity times slope.
1303  !! On exit, equals the streamfunction.
1304  ! Local variables
1305  integer :: k
1306 
1307  real :: b_denom, beta, d1, c1(nk)
1308 
1309  sfn(1) = 0.
1310  b_denom = hn2(2) + c2_h(1)
1311  beta = 1.0 / ( b_denom + c2_h(2) )
1312  d1 = beta * b_denom
1313  sfn(2) = ( beta * hn2(2) )*sfn(2)
1314  do k=3,nk
1315  c1(k-1) = beta * c2_h(k-1)
1316  b_denom = hn2(k) + d1*c2_h(k-1)
1317  beta = 1.0 / (b_denom + c2_h(k))
1318  d1 = beta * b_denom
1319  sfn(k) = beta * (hn2(k)*sfn(k) + c2_h(k-1)*sfn(k-1))
1320  enddo
1321  c1(nk) = beta * c2_h(nk)
1322  sfn(nk+1) = 0.
1323  do k=nk,2,-1
1324  sfn(k) = sfn(k) + c1(k)*sfn(k+1)
1325  enddo
1326 
1327 end subroutine streamfn_solver

2039 !!
2040 !! \section section_gm Thickness diffusion (aka Gent-McWilliams)
2041 !!
2042 !! Thickness diffusion is implemented via along-layer mass fluxes
2043 !! \f[
2044 !! h^\dagger \leftarrow h^n - \Delta t \nabla \cdot ( \vec{uh}^* )
2045 !! \f]
2046 !! where the mass fluxes are cast as the difference in vector streamfunction
2047 !!
2048 !! \f[
2049 !! \vec{uh}^* = \delta_k \vec{\psi} .
2050 !! \f]
2051 !!
2052 !! The GM implementation of thickness diffusion made the streamfunction proportional to the potential density slope
2053 !! \f[
2054 !! \vec{\psi} = - \kappa_h \frac{\nabla_z \rho}{\partial_z \rho}
2055 !! = \frac{g\kappa_h}{\rho_o} \frac{\nabla \rho}{N^2} = \kappa_h \frac{M^2}{N^2}
2056 !! \f]
2057 !! but for robustness the scheme is implemented as
2058 !! \f[
2059 !! \vec{\psi} = \kappa_h \frac{M^2}{\sqrt{N^4 + M^4}}
2060 !! \f]
2061 !! since the quantity \f$\frac{M^2}{\sqrt{N^2 + M^2}}\f$ is bounded between $-1$ and $1$ and does not change sign
2062 !! if \f$N^2<0\f$.
2063 !!
2064 !! Optionally, the method of Ferrari et al, 2010, can be used to obtain the streamfunction which solves the
2065 !! vertically elliptic equation:
2066 !! \f[
2067 !! \gamma_F \partial_z c^2 \partial_z \psi - N_*^2 \psi = ( 1 + \gamma_F ) \kappa_h N_*^2 \frac{M^2}{\sqrt{N^4+M^4}}
2068 !! \f]
2069 !! which recovers the previous streamfunction relation in the limit that \f$ c \rightarrow 0 \f$.
2070 !! Here, \f$c=\max(c_{min},c_g)\f$ is the maximum of either \f$c_{min}\f$ and either the first baroclinic mode
2071 !! wave-speed or the equivalent barotropic mode wave-speed.
2072 !! \f$N_*^2 = \max(N^2,0)\f$ is a non-negative form of the square of the Brunt-Vaisala frequency.
2073 !! The parameter \f$\gamma_F\f$ is used to reduce the vertical smoothing length scale.
2074 !! \f[
2075 !! \kappa_h = \left( \kappa_o + \alpha_{s} L_{s}^2 < S N > + \alpha_{M} \kappa_{M} \right) r(\Delta x,L_d)
2076 !! \f]
2077 !! where \f$ S \f$ is the isoneutral slope magnitude, \f$ N \f$ is the square root of Brunt-Vaisala frequency,
2078 !! \f$\kappa_{M}\f$ is the diffusivity calculated by the MEKE parameterization (mom_meke module) and
2079 !! \f$ r(\Delta x,L_d) \f$ is a function of the local resolution (ratio of grid-spacing, \f$\Delta x\f$,
2080 !! to deformation radius, \f$L_d\f$). The length \f$L_s\f$ is provided by the mom_lateral_mixing_coeffs module
2081 !! (enabled with <code>USE_VARIABLE_MIXING=True</code> and the term \f$<SN>\f$ is the vertical average slope
2082 !! times the Brunt-Vaisala frequency prescribed by Visbeck et al., 1996.
2083 !!
2084 !! The result of the above expression is subsequently bounded by minimum and maximum values, including an upper
2085 !! diffusivity consistent with numerical stability (\f$ \kappa_{cfl} \f$ is calculated internally).
2086 !! \f[
2087 !! \kappa_h \leftarrow \min{\left( \kappa_{max}, \kappa_{cfl}, \max{\left( \kappa_{min}, \kappa_h \right)} \right)}
2088 !! f(c_g,z)
2089 !! \f]
2090 !!
2091 !! where \f$f(c_g,z)\f$ is a vertical structure function.
2092 !! \f$f(c_g,z)\f$ is calculated in module mom_lateral_mixing_coeffs.
2093 !! If <code>KHTH_USE_EBT_STRUCT=True</code> then \f$f(c_g,z)\f$ is set to look like the equivalent barotropic
2094 !! modal velocity structure. Otherwise \f$f(c_g,z)=1\f$ and the diffusivity is independent of depth.
2095 !!
2096 !! In order to calculate meaningful slopes in vanished layers, temporary copies of the thermodynamic variables
2097 !! are passed through a vertical smoother, function vert_fill_ts():
2098 !! \f{eqnarray*}{
2099 !! \left[ 1 + \Delta t \kappa_{smth} \frac{\partial^2}{\partial_z^2} \right] \theta & \leftarrow & \theta \\
2100 !! \left[ 1 + \Delta t \kappa_{smth} \frac{\partial^2}{\partial_z^2} \right] s & \leftarrow & s
2101 !! \f}
